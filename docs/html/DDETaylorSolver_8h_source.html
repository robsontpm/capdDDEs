85  void encloseSolution(
86  TimePointType const& in_t0,
87  SolutionCurveType const& in_curve,
88  TimePointType& out_th,
89  RealType& out_HH,
90  VariableStorageType& out_u,
91  ValueStorageType& out_u_encl,
92  ValueStorageType& out_Phi_coeffs_t0,
93  JacobianStorageType& out_JacPhi_coeffs_t0,
94  ValueStorageType& out_Rem_coeffs_t0,
95  ValueStorageType& out_Y, // this will be the enclosure of the jet of the solution over time [t0, t0+h]
96  ValueStorageType& out_Phi_z, // for possible future use, I assume that there might be more than only value of x(t+h) !
97  JacobianStorageType& out_JacPhi_z, // I need storage for as many as there will be entries in out_Phi_z
98  ValueStorageType& out_Rem_z // I need storage for as many as there will be entries in out_Phi_z
99  )
100  {
101  size_type d = m_map.imageDimension();
102  VectorType zero(d); MatrixType Zero(d, d);
103 
104  RealType h = in_curve.getStep();
105  out_th = in_t0 + in_curve.getStep();
106  out_HH = ecloseStep(h);
107 
108  size_type order;
109  // collecting computation data will make things faster and allow to collect u_encl - needed in rigorous computations.
110  m_map.collectComputationData(in_t0, out_th, out_HH, in_curve, out_u, out_u_encl, order);
111  // order is the final result order of the new representation at the current time
112  // i.e. it will be one higher than the smallest orders at each delay.
113  // then we take into account the possibility that order is too high for the computation to be feasible
114  // order = order > m_maxOrder ? m_maxOrder : order;
115  size_type coeffs_order = order;
116  bool extraOrder = false;
117  if (order > m_maxOrder){
118  // extra rank computation as in old method
119  extraOrder = true;
120  order = m_maxOrder;
121  coeffs_order = m_maxOrder + 1;
122  }
123 
124  if (out_Phi_coeffs_t0.size() != coeffs_order + 1)
125  out_Phi_coeffs_t0.resize(coeffs_order + 1);
126 
127  // make sure that JacPhi can store enough data
128  out_JacPhi_coeffs_t0.resize(coeffs_order + 1);
129  for (auto iJac = out_JacPhi_coeffs_t0.begin(); iJac != out_JacPhi_coeffs_t0.end(); ++iJac)
130  iJac->resize(out_u.size(), Zero);
131 
132  // those will be used for computation of DDECeoficients
133  // uMid - middle point of variables in u
134  // uHull - interval hull of u
135  // uEncl - enclosure of variables in u on whole intervals of length out_HH
136  ValueStorageType uMid, uHull, uEncl;
137 
138  variableToValueMiddle(out_u, uMid);
139  variableToValueHull(out_u, uHull);
140 
141  // compute Jacobian of the numerical method on the hull,
142  // out_Phi_coeffs_t0 will be discarded later (computed only for midpoint)
143  m_map.computeDDECoefficients(in_t0, uHull, out_Phi_coeffs_t0, out_JacPhi_coeffs_t0);
144 
145  // this is to test older codes.
146  // this is version where I compute by interval method (not mean value form)
147  // mean value form is below.
148  VectorType extraOrderAtT0J1(d);
149  if (extraOrder)
150  extraOrderAtT0J1 = out_Phi_coeffs_t0[coeffs_order];
151 
152  // compute coefficients for the middle of the set, overwriting previous out_Phi_coeffs_t0
153  m_map.computeDDECoefficients(in_t0, uMid, out_Phi_coeffs_t0);
154 
155  // now we have Taylor coefficients at t0, we now simply use standard
156  // Taylor method, as in case of ODEs (with the except, that we must
157  // compute roughEnclosure and enclosure of the solution over [t0, t0+h] differently
158 
159  out_Phi_z.clear(); out_Phi_z.resize(1, zero); // there will be only one dependent value, i.e. x(t0+h)
160  out_JacPhi_z.clear(); out_JacPhi_z.resize(1); // so, there will be only one Jacobian entry
161  out_JacPhi_z[0].resize(out_u.size(), Zero); // but it will be dependent on the same variables in u
162 
163  // we explicitely eval jet at t0 as a Taylor series to assure that the
164  // jacobian computed along is valid.
165  // TODO: (NOT URGENT) rewrite as iterators?
166  size_type k = order;
167  while(true){
168  out_Phi_z[0] = out_Phi_coeffs_t0[k] + h * out_Phi_z[0];
169  for (size_type j = 0; j < out_u.size(); ++j){
170  out_JacPhi_z[0][j] = out_JacPhi_coeffs_t0[k][j] + h * out_JacPhi_z[0][j];
171  }
172  if (k == 0) break; else --k; // this prevents from infinite loop in case size_type is unsigned
173  }
174 
175  // there is no remainder for coeffs - the values does not depend
176  // on the rough enclosure on the Xi part of the set. Therefore, we set to zero.
177  out_Rem_coeffs_t0.clear(); out_Rem_coeffs_t0.resize(coeffs_order + 1, zero);
178  out_Rem_z.clear(); out_Rem_z.resize(1, zero);
179  // in out_CoeffsEnclosure we can compute up to order + 1 (incl.)
180  // because we have enclosures on the coefficients up to order order
181  // for all delayed points (Xi is the enclosure on the highest possible order)
182  // Y is the name of the variable in the current paper/notes on extension of the solver.
183  // please note, that rough enclosure is computed inside m_map.collectComputationData(...).
184  out_Y.clear(); out_Y.resize(coeffs_order + 2, zero);
185 
186  // then w propagate it with the map
187  m_map.computeDDECoefficients(in_t0, out_u_encl, out_Y);
188 
189  if (extraOrder){
190  // TODO: (NOT URGENT, RETHINK) this is just to be sure the code is the same as old one, but probably not needed, since expansion of the representation is more natural...
191  // implement extra expansion when order = maxOrder (as in old codes)
192 
193  // THIS IS BAD HACK JUST FOR TEST PURPOSES (move logic in solver to get better results)
194  // HACK STARTS HERE
195  MatrixType extra_C(d, in_curve.storageN0());
196  // compute value of the extra coefficient used later in Xi
197  for (size_type iu = 0; iu < out_u.size(); ++iu){
198  MatrixType C_u = out_u[iu].get_C(); // TODO: (NOT URGENT) get_C copies matrix, now it is safe interface, but for speed rethink it should return reference / pointer?
199  extra_C += out_JacPhi_coeffs_t0[order + 1][iu] * C_u;
200  }
201  MatrixType S = extra_C; S *= 0.;
202  split(extra_C, S);
203  // in this I do not care about extra_B - i will cast it to Vector anyway.
204  VectorType extra_x(d), extra_r(d);
205  extra_x = out_Phi_coeffs_t0[coeffs_order] + out_Rem_coeffs_t0[coeffs_order];
206  split(extra_x, extra_r);
207  split(extra_C, S);
208  extra_r += S * in_curve.get_r0();
209  for (size_type iu = 0; iu < out_u.size(); ++iu)
210  extra_r += (out_JacPhi_coeffs_t0[coeffs_order][iu] * out_u[iu].get_B()) * out_u[iu].get_r();
211 
212  // HACK ENDS HERE
213 
214  // now modify new_Xi to reflect this. We have two versions.
215  // first version of alternate out_Y[order + 1] computed with Interval arythm.
216  VectorType extra_YJ1 = extraOrderAtT0J1 + (order + 2) * out_HH * out_Y[order + 2];
217  // first version of alternate out_Y[order + 1] computed with Interval arythm.
218  VectorType extra_YJ2 = extra_x + (extra_C * in_curve.get_r0()) + extra_r + (order + 2) * out_HH * out_Y[order + 2];
219 
220  try {
221  out_Y[order + 1] = capd::vectalg::intersection(extra_YJ1, out_Y[order + 1]);
222  } catch (...) {
223  throw std::logic_error("CRITICAL: empty intersection of sets that THEORETICALLY ARE PROVED TO BE NON-EMPTY. Please contact authors (robert.szczelina@uj.edu.pl), as the source code could contain some error.");
224  }
225 
226  // we need to clean the extra coefficients from data, so that set.move() is not affected
227  out_Y.pop_back();
228  out_Phi_coeffs_t0.pop_back();
229  out_JacPhi_coeffs_t0.pop_back();
230  out_Rem_coeffs_t0.pop_back();
231  }
232 
233  // we extract the remainder of x(t) over [t0, th]:
234  out_Rem_z[0] = power(h, order + 1) * out_Y[order + 1];
235  }
