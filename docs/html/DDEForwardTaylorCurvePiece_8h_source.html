525  template<typename IteratorType>
526  void setupFromData(IteratorType it, IteratorType itEnd){
527  if (itEnd <= it) throw std::logic_error("DDEForwardTaylorCurvePiece::copyFromData(): cannot copy from empty range.");
528  m_order = itEnd - it - 1;
529  m_dimension = it->dimension(); // we assume dimension of the first element
530  reallocate(storageN0()); // this will unnecessary reallocate R0, rethink...
531  MatrixType eachC(m_dimension, storageN0()); // to make sets of good shape to N0. The matrix = 0, so it should be set later by end user.
532  std::cout << "storageN0-piece" << storageN0() << std::endl;
533  for (auto jk = beginJet(); it != itEnd; ++it, ++jk){
534  try{ dimCheck(&(*it)); } catch (std::logic_error& e) { throw rethrow("DDEForwardTaylorCurvePiece::copyFromData(): incompatible dimension.", e); }
535  try{*jk = SetType(*it, eachC, *m_r0);} catch (std::logic_error& e) { throw rethrow("DDEForwardTaylorCurvePiece::copyFromData(): SetType constructor error.", e); }
536  }
537  // previously, in SetType(..., eachC, *m_r0) we used a copy of the m_r0. Now we make it shared.
538  try{ updateCommonR0(); } catch (std::logic_error& e) { throw rethrow("DDEForwardTaylorCurvePiece::copyFromData(): updateCommonR0 error.", e); }
539  }

547  template<typename XItSpec, typename CItSpec, typename BItSpec, typename RItSpec>
548  void setupFromData(
549  XItSpec xit, XItSpec xend,
550  CItSpec Cit, CItSpec Cend,
551  VectorType* r0,
552  BItSpec Bit, BItSpec Bend,
553  BItSpec invBit, BItSpec invBend,
554  RItSpec rit, RItSpec rend,
555  VectorType* Xi)
556  {
557  deallocate();
558  if (xend <= xit || Cend <= Cit || Bend <= Bit || rend <= rit)
559  throw std::logic_error("DDEForwardTaylorCurvePiece::copyFromData(): cannot copy from empty range.");
560  m_order = xend - xit - 1;
561  if (Cend - Cit <= m_order || Bend - Bit <= m_order || rend - rit <= m_order)
562  throw std::logic_error("DDEForwardTaylorCurvePiece::copyFromData(): not enough data to construct object.");
563  m_dimension = xit->dimension();
564  m_Xi = Xi;
565  m_r0 = r0;
566  // we only allocate Jet, as r0 and Xi are already present as pointers.
567  allocateJet();
568  for (auto jk = beginJet(); jk != endJet(); ++jk, ++xit, ++Cit, ++Bit, ++invBit, ++rit){
569  try{ dimCheck(&(*xit)); } catch (std::logic_error& e) { throw rethrow("DDEForwardTaylorCurvePiece::copyFromData(): incompatible dimension in x.", e); }
570  try{ dimCheck(&(*rit)); } catch (std::logic_error& e) { throw rethrow("DDEForwardTaylorCurvePiece::copyFromData(): incompatible dimension in r.", e); }
571  try{ dimCheckC(&(*Cit)); } catch (std::logic_error& e) { throw rethrow("DDEForwardTaylorCurvePiece::copyFromData(): incompatible dimension in C.", e); }
572  try{ dimCheckB(&(*Bit)); } catch (std::logic_error& e) { throw rethrow("DDEForwardTaylorCurvePiece::copyFromData(): incompatible dimension in B.", e); }
573  try{ dimCheckB(&(*invBit)); } catch (std::logic_error& e) { throw rethrow("DDEForwardTaylorCurvePiece::copyFromData(): incompatible dimension in invB.", e); }
574  *jk = SetType(*xit, *Cit, r0, *Bit, *invBit, *rit);
575  }
576  // no need for updateCommonR0 - creation of sets already done that
577  }
