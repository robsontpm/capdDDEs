388  template<typename FunctionalSpec>
389  Class dt(FunctionalSpec const& f, size_type n = 1) const {
390  if (n == 0) { return *this; } // no need to compute anything
391  Class result(m_grid, pastTime(), dimension());
392  for (auto ijet = begin(); ijet != end(); ++ijet)
393  result.addPiece((*ijet)->dt(n));
394  if (n == 1){
395  // faster and simplified than the case n > 1
396  DataType v = f(*this);
397  result.setValueAtCurrent(v);
398  }else{
399  typedef typename FunctionalSpec::ValueStorageType ValueStorageType;
400  ValueStorageType coeffs;
401  f.computeDDECoefficients(*this, coeffs);
402  DataType v = coeffs[n-1];
403  while (n > 0) v *= n--; // v = x^(n)/n!, so we need to multiply by n!
404  result.setValueAtCurrent(v);
405  }
406  return result;
407  }

544  template<typename DynSysSpec>
545  void epsilonShift(
546  DynSysSpec const& solver,
547  TimePointType const& at_t0, RealType const& epsilon,
548  Class& out_result) const {
549  // TODO: (FUTURE) this is only forward epsilon step. Implement backward step later
550  // TODO: (FUTURE) use solver and history to recompute Xi over shorter intervals t_i + [0, \epsi], when computing coefficients.
551  // TODO: (NOT URGENT) add checks for sanity of the input before proceeding.
552  TimePointType how_far = out_result.t0() - out_result.pastTime();
553  auto ijet_out = out_result.begin();
554  auto ijet_this = this->at(at_t0 - how_far);
555  for (; ijet_out != out_result.end(); ++ijet_out, ++ijet_this){
556  (**ijet_this).evalAtDelta(epsilon, (**ijet_out)[0]);
557  size_type order = (**ijet_out).order();
558  for (size_type k = 1; k <= order; ++k){
559  (**ijet_this).evalCoeffAtDelta(k, epsilon, (**ijet_out)[k]);
560  }
561  }
562  // we assume this extra jet is already computed,
563  // otherwise, we would need to call solver to extend solution - we do not want this.
564  (**ijet_this).evalAtDelta(epsilon, out_result.m_valueAtCurrent);
565  }
