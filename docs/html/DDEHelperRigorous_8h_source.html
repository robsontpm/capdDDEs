231  Solution dataToSolution(CoordinateSystem const& coords, Vector dx, Vector r0, Vector Xi) const {
232  Vector rr = r0;
233  capd::vectalg::split(rr, r0);
234  auto tau = m_grid.point(m_p);
235  auto t_0 = m_grid.point(0);
236  Solution X(m_grid, -tau, t_0, m_n, Vector(DIMENSION));
237  X.set_x(coords.reference + coords.coords * dx + coords.coords * rr);
238  X.set_Cr0(coords.coords, r0);
239  X.set_Xi(Xi);
240  return X;
241  }

365  Solution timemap(Solution& X, const size_type& steps, const Real& epsilon = 0.){
366  auto tau = m_grid.point(m_p);
367  DDEq dde(Eq(m_params), tau);
368  Solver solver(dde, m_maxOrder);
369 
370  Solution TX(X.subcurve(0, m_p)); TX *= 0.;
371  size_type count = epsilon != 0. ? steps + 1 : steps;
372  for (size_type step = 0; step < count; ++step) X.move(solver);
373  if (epsilon != 0.){
374  X.epsilonShift(solver, epsilon, TX);
375  } else {
376  X.subcurve(X.length() - m_p).reduce(TX);
377  }
378  return TX;
379  }

381  Solution timemap(
382  CoordinateSystem const& in_coords,
383  Vector const& dx, Vector const& r0, Vector const& Xi,
384  const size_type& steps, const Real& epsilon,
385  CoordinateSystem const& out_coords,
386  Vector& Tdx, Vector& Tr0, Vector& TXi){
387  Solution X = dataToSolution(in_coords, dx, r0, Xi);
388 
389  auto tau = m_grid.point(m_p);
390  DDEq dde(Eq(m_params), tau);
391  Solver solver(dde, m_maxOrder);
392 
393  // TODO: (NOT URGENT) refactor: DRY, use timemap(Solution, steps, epsilon) implementation
394  Tdx = Vector(M());
395  Solution TX(X); TX *= 0.;
396  size_type count = epsilon != 0. ? steps + 1 : steps;
397  for (size_type step = 0; step < count; ++step) X.move(solver);
398  if (epsilon != 0.){
399  X.epsilonShift(solver, epsilon, TX);
400  } else {
401  X.subcurve(X.length() - m_p).reduce(TX);
402  }
403  // we compute transformation: ref + C^-1 * (PX - ref)
404  Tdx = (out_coords.inverseCoords * (TX.get_x() - out_coords.reference)) + // shift middle by reference and then apply coords change on the remainder
405  (out_coords.inverseCoords * TX.get_C()) * TX.get_r0() + // apply coords change to Cr0 part
406  (out_coords.inverseCoords * TX.get_B()) * TX.get_r(); // apply coords change to Br part
407  capd::vectalg::split(Tdx, Tr0);
408  TXi = TX.get_Xi();
409  return X; // X contains full trajectory
410  }

422  Solution poincare(
423  Section section,
424  capd::poincare::CrossingDirection crossing_direction,
425  Solution const& X,
426  Real& reachTime, size_type& steps, Real& epsilon,
427  Solution& PX){
428  if (X.length() != m_p){
429  std::ostringstream info;
430  info << "RigorousHelper::poincare(section, X, ...): ";
431  info << "X must be of length exactly " << m_p << ", is of length " << X.length() << std::endl;
432  throw std::logic_error(info.str());
433  }
434 
435  auto tau = m_grid.point(m_p);
436  DDEq dde(Eq(m_params), tau);
437  Solver solver(dde, m_maxOrder);
438  PoincareMap pm(solver, section);
439  pm.setDirection(crossing_direction);
440  pm.setRequiredSteps(m_reqSteps);
441  pm.setMaxSteps(m_maxSteps);
442 
443  Solution TX = X;
444  PX *= 0.;
445  pm(TX, reachTime, PX);
446  epsilon = pm.getLastEpsilonTime();
447  steps = TX.length() - m_p - 1; // TODO: this is not accurate... but for now it must suffice
448  return TX; // TX contains full trajectory
449  }
