200  void startNewParam(const std::string &param) {
201  help_oss << "\n";

222  template<typename T>
223  friend ArgumentParser& operator<<(ArgumentParser &out, const T &item) {
224  std::ostringstream tmp; tmp << item;
225  char stop = 1;
226  std::string tmps = tmp.str() + stop;
227  std::string word = "";
228  for (auto& c : tmps){
229  // std::cout << "c|" << c << "|word|" << word << std::endl;
230  if (word.length() + out.current_pad.length() >= out.MAX_LENGTH){
231  // word is too long to put in a one line, split...
232  out.help_oss << word; word = "";
233  out.current_col = out.current_pad.length();
234  }else if (out.current_col >= out.MAX_LENGTH){
235  out.help_oss << "\n" << out.current_pad;
236  out.current_col = out.current_pad.length();
237  } else if (c == ' ' || c == '\t') {
238  out.help_oss << word; word = "";
239  out.help_oss << " "; ++out.current_col;
240  continue;
241  } else if (c == '\n' || c == '\r') {
242  out.help_oss << word; word = "";
243  out.help_oss << "\n" << out.current_pad;
244  out.current_col = out.current_pad.length();
245  continue;
246  } else if (c == stop) {
247  out.help_oss << word;
248  continue;
249  }
250  word += c; ++out.current_col;
251  }
252  return out;
253  }
