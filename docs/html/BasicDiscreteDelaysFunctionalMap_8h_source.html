158  const TimePointType& t0, const TimePointType& th, //input
159  const RealType& dt, const CurveType& x, // input

161  size_type& out_admissible_order) const { // output
162  // TODO: (NOT URGENT) DRY with the other collectComputationData() method.
163  out_u.clear();
164  // we collect jets and compute admissible order
165  // TODO: (NOT URGENT) rewrite using pointers to Jets (will be faster)!
166  out_admissible_order = 40;
167  std::vector<JetType> jets;
168  for (auto tau = begin(); tau != end(); ++tau){
169  jets.push_back(x.jet(t0 - *tau));
170  size_type k = jets.back().order();
171  if (k + 1 < out_admissible_order) out_admissible_order = k + 1;
172  }
173  // Then we fill in at u[0] the value at current time (always present in u, by assumption)
174  out_u.push_back(x.getValueAtCurrent());
175  // and then all jets, order by order (in this way it is easier to fill AutoDiff later)
176  for (size_type k = 0; k < out_admissible_order; ++k)
177  for (auto jit = jets.begin(); jit != jets.end(); ++jit)
178  out_u.push_back((*jit)[k]);

182  virtual void computeDDECoefficients(
183  const RealType& t0, const ValueStorageType& u, // input
184  ValueStorageType& coeffs) const { // output
185  // we assume u[0] is always present, and all other u[i] has the same
186  checkDimension(u[0]);
187  // we assume that coeffs has a proper size for the computation.
188  // for example, set when user took data u with collectComputationData()
189  size_type resultOrder = coeffs.size() - 1;
190 
191  typedef fadbad::T<ScalarType> AutoDiffTScalar; // Auto Diff T (Taylor) scalar type
192  typedef typename VectorType::template rebind<AutoDiffTScalar>::other TADVectorType;
193  TADVectorType args(dimension());
194  // propagate arguments in a given order - first value at t0
195  // then jet of \tau_1, then \tau_2, etc...
196  // this is change from the initial paper!, where I have done delay terms first
197  // this is coherent with what is in the general DDE literature though...
198  // we assume this order in u (because we produce itd this way in collectComputationData()!).
199  auto current_u = u.begin();
200  auto iarg = args.begin();
201  for (size_type idelay = 0; idelay <= delaysCount(); ++idelay, ++current_u)
202  for (auto iu = current_u->begin(); iu != current_u->end(); ++iu, ++iarg)
203  (*iarg)[0] = *iu;
204 
205  // we are ready to go for the recurrent relation on coefficients.
206  coeffs[0] = u[0]; // by definition, see paper
207  for (size_type k = 1 ; k <= resultOrder; ++k){
208  // eval Autodiff routine so that we would be able to extract v[k-1] = F^{[k-1]}(...)
209  TADVectorType v(imageDimension());
210  m_map(t0, args, v);
211  // propagate AD jets and output.
212  // this will be F_k from the latest paper on DDE integration.
213  VectorType Fk(imageDimension());
214  auto iarg = args.begin();
215  for (size_type i = 0; i < imageDimension(); ++i, ++iarg){
216  v[i].eval(k-1);
217  Fk[i] = v[i][k-1];
218  // propagate computed recurrent formula into args with proper scaling!
219  (*iarg)[k] = Fk[i] / ScalarType(k);
220  }
221  // this is the right formula from the last paper!
222  coeffs[k] = Fk / ScalarType(k);
223  // do not propagate anything else if we are past the requestedOrder (it might cause errors)
224  if (k == resultOrder) break;
225  // propagate higher order coefficients in the past to the args Vector.
226  // the recurrent Taylor formula is already there.
227  for (size_type idelay = 1; idelay <= delaysCount(); ++idelay, ++current_u){
228  for (auto iu = current_u->begin(); iu != current_u->end(); ++iu, ++iarg)
229  (*iarg)[k] = *iu;
230  }
231  }

237  virtual void computeDDECoefficients(
238  const RealType& t0, const ValueStorageType& u, // input
239  ValueStorageType& coeffs, JacobianStorageType& Du) const { // output
240  // we assume u[0] is always present, and all other u[i] has the same
241  checkDimension(u[0]);
242  // we assume that coeffs has a proper size for the computation.
243  // for example, set when user took data u with collectComputationData()
244  size_type resultOrder = coeffs.size() - 1;
245  // d = space dimension of a single variable in the map / input data in ValueStorageType (must be the same).
246  size_type d = this->imageDimension();
247  size_type numDelays = delaysCount();
248  // those are common matrices of dimension M(d, d),
249  MatrixType Id(d, d); Id.setToIdentity();
250  MatrixType Zero(d, d);
251 
252  // This is obvious
253  size_type partialsCount = u.size();
254  // in partialsDimension we keep how many actual coefficients from all of the Jets used to
255  // compute rhs of the equation. As we are computing jets up to coeffs.order() (incl.).
256  // then we are going to use only coefficients of all jets up to coeffs.order() - 1 (incl).
257  // Thus, the formula. Please note in all those computations we have 0,..,resultOrder-1
258  // Taylor coefficients in any jet used. Therefore the number of Taylor coeffs at each
259  // jet used is resultOrder. This is trivial, but I want this to be reiterated for readability.
260  size_type partialsDimension = d * u.size();
261  // we make room to store Jacobian matrices. Each matrix will be M(d,d).
262  // Du is shorthand for $D_u coeffs$
263  // in Du[i][j] we will have \partial coeffs[i]
264  Du.clear(); Du.resize(resultOrder + 1);
265  for (auto iDu = Du.begin(); iDu != Du.end(); ++iDu)
266  iDu->resize(partialsCount);
267 
268  typedef fadbad::F<ScalarType> FADScalar; // Auto Diff Forward scalar type (i.e. w.r.t. initial coefficients)
269  typedef fadbad::T<FADScalar> TFADScalar; // Auto Diff Taylor type with option to compute derivatives wrt initials
270  typedef typename VectorType::template rebind<TFADScalar>::other TFADVectorType;
271  // this is Automatic Differentiation both w.r.t t but also w.r.t. coefficients.
272  // we will use it to automatically eval the r.h.s. of the equation with all
273  // partial derivatives when recursively compute Taylor coefficients at current t0
274  TFADVectorType args(dimension());
275 
276  // we will successively fill-in args as the recursion will go on
277  auto current_u = u.begin();
278  auto iarg = args.begin();
279  // this will hold current number of parameter (among 0,.., partialsDimension-1)
280  // we need those for AutoDiff types (they use this kind of numbering)
281  size_type partialAD = 0;
282  // this will hold current number of variable in u (among 0,..,partialsCount-1, each variable is d dimensional)
283  // we need those for our purpose (to match scalar variables to appropriate entries in vector/matrices)
284  size_type partialu = 0;
285  // we fill-in initial values for the recurrence, i.e. the value at t0 stored
286  // in u[0] and values at values in the past, at given delays.
287  // We mark each of those values as dependent only on self. This will give us
288  // possibility to automatically compute dependence of the formulas on them
289  // i.e. the value \frac{\partial coeffs}{\partial u[partialu]}
290  // Note: here k = 0, and we iterate idelay from 0. Later, it will be from 1. See docs there.
291  for (size_type idelay = 0; idelay <= numDelays; ++idelay, ++current_u, ++partialu){
292  for (auto iu = current_u->begin(); iu != current_u->end(); ++iarg, ++iu, ++partialAD){
293  // (*iarg)[0] is [0]-th order Taylor coefficient
294  (*iarg)[0] = *iu;
295  // and we mark it as dependent only on self among all 0,..,partialsDimension-1 variables
296  // (i.e. \frac{\partial{(*iarg)[0]}}{\partial{(*iarg)[0]}} = 1, and 0 otherwise)
297  (*iarg)[0].diff(partialAD, partialsDimension);
298  }
299  }
300 
301  // we compute coeffs[0] and derivatives...
302  coeffs[0] = u[0]; // coeffs[0] = x(t0) by definition of (see paper)
303  Du[0][0] = Id; // \frac{Dcoeffs_{[0]}}{Dx(t0)} = Id (as a matrix M(d, d)) (since coeffs[0] = x(t0))
304  // Next... \frac{Dcoeffs_{[0]}}{{Djet_m}_{[mk]}} = 0 (as a matrix M(d, d)) (since coeffs[0] = x(t0))
305  for (size_type par = 1; par < partialsCount; ++par)
306  Du[0][par] = Zero;
307 
308  // then we compute higher order coeffs, with recurrent Taylor formula (see paper)
309  for (size_type k = 1; k <= resultOrder; ++k){
310  // eval AutoDiff routine so we would be able to extract v[.][k-1] = F^{[k-1]}(...)
311  // together with partial derivatives with respect to coefficients in the past.
312  TFADVectorType v(imageDimension());
313  m_map(t0, args, v);
314 
315  // propagate AD jets and output.
316  VectorType Fk(imageDimension());
317  auto iarg = args.begin();
318  for (size_type i = 0; i < imageDimension(); ++i, ++iarg){
319  v[i].eval(k-1);
320  Fk[i] = v[i][k-1].val(); // here we need to .value(), to convert from FAD to VectorType
321  (*iarg)[k] = v[i][k-1] / ScalarType(k); // here we need to propagate things with all partial derivatives! So we copy from v (appropriate AutoDiff Type)
322  }
323 
324  // we output the value, properly scaled
325  coeffs[k] = Fk / ScalarType(k);
326  // and partial derivatives w.r.t all previous coeffs.
327  MatrixType parDu(d, d);
328  for (size_type par = 0; par < partialu; ++par){
329  // we copy the appropriate partial derivatives into a local Matrix, which will be later
330  // Du[k][par] = \frac{\partial coeffs[k]}{\partial u[par]}
331  for (size_type i = 0; i < d; ++i) // we use formula par*d+j
332  for (size_type j = 0; j < d; ++j) // to obtain the local index
333  parDu[i][j] = v[i][k-1].d(par * d + j) / ScalarType(k); // among all partialsCount variables
334  Du[k][par] = parDu;
335  }
336  // mark that this order coeff does not depend on higher order u's
337  for (size_type par = partialu; par < partialsCount; ++par)
338  Du[k][par] = Zero;
339 
340 
341  // this if is to ensure that we do not copy, if there is no more values left in u.
342  if (k == resultOrder) break;
343 
344  // We have now propagated dependence of the solution on the value at
345  // present time t0 to Du. Now, finally, we propagate higher order
346  // coefficients in the past to args for the next level of recurrence.
347  for (size_type idelay = 1; idelay <= numDelays; ++idelay, ++current_u, ++partialu){
348  for (auto iu = current_u->begin(); iu != current_u->end(); ++iarg, ++iu, ++partialAD){
349  // (*iarg)[0] is [0]-th order Taylor coefficient
350  (*iarg)[k] = *iu;
351  // and we mark it as dependent only on self among all 0,..,partialsDimension-1 variables
352  // (i.e. \frac{\partial{(*iarg)[0]}}{\partial{(*iarg)[0]}} = 1, and 0 otherwise)
353  (*iarg)[k].diff(partialAD, partialsDimension);
354  }
355  }
356  // TODO: (FUTURE) rewrite on iterators only, it is possible for sure!
357  // TODO: (FUTURE) in such a case probably, I would not need many of the *Count variables.
358 
359  // now we are ready for the next level of recursion.
360  } // for k loop
361  } // function computeDDECoefficients
