151  if (!x) throw std::logic_error("BasicDoubleton::__construct__: x cannot be NULL");

155  BB = *B;
156  } else {
157  BB = MatrixType(x->dimension(), x->dimension());
158  BB.setToIdentity();
159  }
160  if (r0)
161  setupFromData(x, C, *r0, BB, r ? *r : VectorType(x->dimension()));
162  else
163  setupFromData(x, C, VectorType(0), BB, r ? *r : VectorType(x->dimension()));
164  this->sanityCheck("::__construct__");
165  }

290  void sanityCheck(std::string const& what = "") const {
291  if (storageN0() != m_C.numberOfColumns()) {
292  std::ostringstream extraInfo; extraInfo << storageN0() << " vs " << m_C.numberOfColumns();
293  throw std::domain_error("BasicDoubleton"+what+": C and r0 have incompatible dimensions: " + extraInfo.str());
294  }
295  if (dimension() != m_B.numberOfRows() || dimension() != m_B.numberOfColumns()) {
296  std::ostringstream info;
297  info << "BasicDoubleton" << what << ": B has incompatible dimensions.";
298  info << "was " << dimension() << ", expected " << m_B.numberOfRows() << ".";
299  throw std::domain_error(info.str());
300  }
301  if (dimension() != m_C.numberOfRows()) {
302  std::ostringstream extraInfo; extraInfo << dimension() << " vs " << m_C.numberOfRows();
303  throw std::domain_error("BasicDoubleton"+what+": C and x has incompatible dimensions: " + extraInfo.str());
304  }
305  if (dimension() != m_r.dimension()) {
306  std::ostringstream extraInfo; extraInfo << dimension() << " vs " << m_r.dimension();
307  throw std::domain_error("BasicDoubleton"+what+": r has incompatible dimensions." + extraInfo.str());
308  }
