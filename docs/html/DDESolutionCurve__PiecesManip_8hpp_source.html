47 template<typename SetSpec>
48 DDESolutionCurve<SetSpec>&
49 DDESolutionCurve<SetSpec>::addPiece(CurvePieceType* newPiece, bool passOwnership){
50  if (newPiece->storageN0() != storageN0())
51  throw std::logic_error("DDESolutionCurve::addPiece(): the new piece has different N_0.");
52  if (!passOwnership)
53  newPiece = new CurvePieceType(*newPiece);
54  m_pieces.push_back(newPiece);
55  m_pieceOwner.push_back(passOwnership);
56  newPiece->setT0(currentTime());
57  newPiece->set_r0(m_r0);
58  m_t_current += m_grid.point(1);
59  return *this;
60 }

87 template<typename SetSpec>
88 DDESolutionCurve<SetSpec>&
89 DDESolutionCurve<SetSpec>::setValueAtCurrent(const SetType& value){
90  if (value.storageN0() != storageN0())
91  throw std::logic_error("DDESolutionCurve::setValueAtCurrent(set): provided value has incompatible N0 dimension.");
92  if (value.dimension() != dimension())
93  throw std::logic_error("DDESolutionCurve::setValueAtCurrent(set): provided value has incompatible dimension.");
94  this->m_valueAtCurrent = value;
95  if (!value.common_r0(m_r0))
96  *m_r0 = intervalHull(*m_r0, value.get_r0());
97  return *this;
