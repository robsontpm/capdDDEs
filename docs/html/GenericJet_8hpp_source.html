124 GenericJet<TimePointSpec, DataSpec, VectorSpec, MatrixSpec, isInterval>::setupCoeffs( // func decl
125  const_iterator it, const_iterator itEnd, size_type overrideOrder // params
126 ){
127  if (it == NULL || itEnd == NULL || itEnd < it)
128  throw std::logic_error("GenericJet::setupCoeffs(begin*, end*): no data given (either one of ptr is NULL or begin > end)");
129  deallocateCoeffs();
130  m_dimension = it->dimension();
131  m_order = overrideOrder ? overrideOrder : (itEnd - it - 1);
132  allocateCoeffs();
133  auto ic = this->begin();
134  for (size_type i = 0; it != itEnd && i <= m_order; ++it, ++ic, ++i){
135  if (it->dimension() != m_dimension){
136  std::ostringstream info;
137  info << "GenericJet::setupCoeffs(begin*, end*): incompatible dimension of coeff, ";

141  throw std::logic_error(info.str());
142  }
143  *ic = *it;
144  }
