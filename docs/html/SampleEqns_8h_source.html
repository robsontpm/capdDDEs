124  template<typename RealSpec, typename InVectorSpec, typename OutVectorSpec>
125  void operator()(const RealSpec& t, const InVectorSpec x, OutVectorSpec& fx) const {
126  typedef typename OutVectorSpec::ScalarType OutScalarType;
127  // x[0] is value at a current time, x[1] is at t-\tau
128  // we use identity:
129  // (x(t-tau))^n = (e^log(x(t-tau))) ^ n = e^(log(x(t)-tau) * n)
130  OutScalarType x1_to_n = exp(log(x[1]) * OutScalarType(n));
131  fx[0] = -gamma * x[0] + ( (beta * x[1]) / (1.0 + x1_to_n) );
132  }
133 
134  static std::string show(){
135  return "Mackey-Glass equation: $x'(t) = -gamma x(t) + \\beta {{ x(t-\\tau) }\\over{ 1 + (x(t-\\tau))^n }}$.";
136  }

261  void reinitialize(){
262  if (A.numberOfRows() != B.numberOfRows()) throw std::logic_error("LinearMap: A and B has different number of rows");
263  if (A.numberOfColumns() != B.numberOfColumns()) throw std::logic_error("LinearMap: A and B has different number of cols");
264  if (A.numberOfColumns() != A.numberOfRows()) throw std::logic_error("LinearMap: A and B must be square");

266  AB = MatrixType(imageDimension(), dimension());
267  for (size_type i = 0; i < imageDimension(); ++i){

423  template<typename RealSpec, typename InVectorSpec, typename OutVectorSpec>
424  void operator()(const RealSpec& t, const InVectorSpec x, OutVectorSpec& fx) const {
425  // "classic" rossler
426  fx[0] = -(x[1]+x[2]);
427  fx[1] = x[0]+a*x[1];
428  fx[2] = b+x[2]*(x[0]-c);
429 
430  // "delayed" term
431  fx[0] += epsi * (-(x[4]+x[5]));
432  fx[1] += epsi * (x[3]+a*x[4]);
433  fx[2] += epsi * (b+x[5]*(x[3]-c));
434  }
