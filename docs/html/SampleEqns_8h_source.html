123  template<typename RealSpec, typename InVectorSpec, typename OutVectorSpec>
124  void operator()(const RealSpec& t, const InVectorSpec x, OutVectorSpec& fx) const {
125  typedef typename OutVectorSpec::ScalarType OutScalarType;
126  // x[0] is value at a current time, x[1] is at t-\tau
127  // we use identity:
128  // (x(t-tau))^n = (e^log(x(t-tau))) ^ n = e^(log(x(t)-tau) * n)
129  OutScalarType x1_to_n = exp(log(x[1]) * OutScalarType(n));
130  fx[0] = -gamma * x[0] + ( (beta * x[1]) / (1.0 + x1_to_n) );
131  }
132 
133  static std::string show(){
134  return "Mackey-Glass equation: $x'(t) = -gamma x(t) + \\beta {{ x(t-\\tau) }\\over{ 1 + (x(t-\\tau))^n }}$.";
135  }

137 protected:
138  ParamType beta;
139  ParamType gamma;
140  ParamType n;

260  void reinitialize(){
261  if (A.numberOfRows() != B.numberOfRows()) throw std::logic_error("LinearMap: A and B has different number of rows");
262  if (A.numberOfColumns() != B.numberOfColumns()) throw std::logic_error("LinearMap: A and B has different number of cols");
263  if (A.numberOfColumns() != A.numberOfRows()) throw std::logic_error("LinearMap: A and B must be square");
264  if (A.numberOfColumns() != b.dimension()) throw std::logic_error("LinearMap: b must be compatible with A and B");
265  AB = MatrixType(imageDimension(), dimension());
266  for (size_type i = 0; i < imageDimension(); ++i){
267  for (size_type j = 0; j < imageDimension(); ++j){
268  AB[i][j] = A[i][j];
269  AB[i][j+imageDimension()] = B[i][j];

422  template<typename RealSpec, typename InVectorSpec, typename OutVectorSpec>
423  void operator()(const RealSpec& t, const InVectorSpec x, OutVectorSpec& fx) const {
424  // "classic" rossler
425  fx[0] = -(x[1]+x[2]);
426  fx[1] = x[0]+a*x[1];
427  fx[2] = b+x[2]*(x[0]-c);
428 
429  // "delayed" term
430  fx[0] += epsi * (-(x[4]+x[5]));
431  fx[1] += epsi * (x[3]+a*x[4]);
432  fx[2] += epsi * (b+x[5]*(x[3]-c));
433  }

436  return "Rossler Toy Delay Model: $x'(t) = f(x(t)) + \\epsilon f(x(t-\\tau))$, where $f$ is r.h.s of classic Rossler ODE in 3D with a,b,c parameters.";
