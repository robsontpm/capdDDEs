48 DDESolutionCurve<SetSpec>& DDESolutionCurve<SetSpec>::move(DynSysSpec& solver){
49  typedef typename DynSysSpec::VariableStorageType Variables;
50  typedef typename DynSysSpec::JacobianStorageType Jacobians;
51  typedef typename DynSysSpec::ValueStorageType Values;
52  Variables u; Values u_encl;
53  Jacobians D_uPhi_j0, D_uPhi_z;
54  Values Phi_z, Rem_z;
55  Values Phi_j0, Rem_j0, Y;
56  TimePointType t_h = m_grid.point(0);
57  RealType HH;
58  // we call solver to enclose all the data needed in a common data structures
59  // for ODEs it was a lot easier, as it was exactly known, what are the
60  // values used in the computations were, it was just the value of the solution
61  // at time t0. Now, we do not know what values the rhs of the equation uses
62  // because it might have a lot of delays, or integrals, or whatever
63  // we trust solver to extract all the relevant data for us.
64  // but we need to pass him a mutable storages that can hold a lot
65  // of Vectors / Matrices. The meaning of u, D_uPhi_j, etc. is
66  // described in the docs of solver and in the new notes / paper.
67  solver.encloseSolution(
68  *this, // this is the only input. rest of variables is output.
69  t_h, HH, // t_h is the new TimePoint of new jet j0, HH is [0, h]
70  u, u_encl, // u is value at current time of variables used in computation, u_encl is over current time + HH
71  Phi_j0, D_uPhi_j0, Rem_j0, Y, // corresponds to the expansion at current time
72  Phi_z, D_uPhi_z, Rem_z // corresponds to the new value at new time (after step), corresponds to result set in ODE setting
73  );
74  m_lastEnclosure = u_encl[0];
75  // now, implement Lohners algorithm with the data provided...
76 
77  // this is the order of the resulting Jet at t0
78  size_type order = Phi_j0.size() - 1;
79  size_type uCount = u.size();
80  size_type d = dimension();
81  size_type N0 = storageN0();
82  // for clarity of writing
83  VectorType const& r0 = *m_r0;
84 
85  // Rem_j0 == 0 here (no remainder from numerical method, as the j0
86  // is computed from rhs of solution with AD),
87  // but we keep track of it in computations nevertheless, if there will be
88  // in future some other algorithm that returns some overestimates
89  MatrixType Zero(d, N0);
90  MatrixType Id(d, d); Id.setToIdentity();
91  // new_ will indicate part of the new Jet to be added to solution
92  // new_C_j0 is a collection of C matrices for sets defining jet at current time
93  // (therefore we have order + 1 of them).
94  // _z is the value at current time (i.e. 0-th coefficient of a new jet,
95  // it corresponds to what is the result set in ODE).
96  std::vector<MatrixType> new_C_j0(order + 1, Zero);
97  MatrixType new_C_z = Zero;
98  MatrixType S = Zero;
99  // WARNING: in the loop below we assume that r0 is common for all
100  // WARNING: sets in u! Otherwise it would be incorrect!
101  for (size_type iu = 0; iu < uCount; ++iu){
102  MatrixType C_u = u[iu].get_C(); // TODO: (FUTURE) get_C copies matrix, now it is safe interface, but for speed rethink it should return reference / pointer?
103  new_C_z += D_uPhi_z[0][iu] * C_u;
104  for (size_type k = 0; k <= order; ++k)
105  new_C_j0[k] += D_uPhi_j0[k][iu] * C_u;
106  }
107 
108  // this will be commonly used by all QR computations, so we store it in tmp variable
109  VectorType u0r = u[0].get_r();
110 
111  // center the value at the current time
112  VectorType new_z = Phi_z[0] + Rem_z[0];
113  VectorType new_r(d); split(new_z, new_r);
114  // we do the QR decomposition with B chosen as the matrix of 0-th variable in u (current time value)
115  // this way we reproduce the algorithm from ODEs (in case there are no delays)
116  MatrixType new_B_z = (D_uPhi_z[0][0] * u[0].get_B());
117  MatrixType new_invB_z(d, d);
118  this->Policy::computeBinvB(new_B_z, new_invB_z, u0r);
119  new_r = new_invB_z * new_r;
120  for (size_type iu = 0; iu < uCount; ++iu)
121  new_r += (new_invB_z * D_uPhi_z[0][iu] * u[iu].get_B()) * u[iu].get_r();
122  // next, incorporate the remains of C * r0
123  // new_C_z is point matrix, while we can put
124  // ((Jac * old_C) - m(Jac * old_C)) * r0 = SC * r0 into remainder!
125  MatrixType SC = Zero;
126  split(new_C_z, SC);
127  new_r += (new_invB_z * SC) * r0;
128 
129  // then we need to redo this for j0 (new Jet at current time)
130  std::vector<MatrixType> new_B_j0(order + 1, Id);
131  std::vector<MatrixType> new_invB_j0(order + 1, Id);
132  // here I cannot change so simply the order of loops, so I leave it outside for Phi_z
133  // it should not be so important, as C matrix in a set might be large
134  // (big number of columns), whereas B matrix is simply M(d, d) (small).
135  for (size_type k = 0; k <= order; ++k){
136  // following two lines centers the set, important!
137  // Phi_j0 will be point set, and Rem_j0 an interval set centered at 0
138  Phi_j0[k] += Rem_j0[k];
139  split(Phi_j0[k], Rem_j0[k]);
140 
141  new_B_j0[k] = (D_uPhi_j0[k][0] * u[0].get_B());
142  this->Policy::computeBinvB(new_B_j0[k], new_invB_j0[k], u0r);
143 
144  Rem_j0[k] = new_invB_j0[k] * Rem_j0[k];
145  for (size_type iu = 0; iu < uCount; ++iu)
146  Rem_j0[k] += (new_invB_j0[k] * D_uPhi_j0[k][iu] * u[iu].get_B()) * u[iu].get_r();
147 
148  // next, incorporate the remains of C * r0
149  // new_C[k] is point matrix, while we can put
150  // ((Jac * old_C) - m(Jac * old_C)) * r0 = S * r0 into remainder!
151  split(new_C_j0[k], S);
152  Rem_j0[k] += (new_invB_j0[k] * S) * r0;
153  }
154 
155  // now compute Xi (i.e. copy value form Jet enclosure at given order)
156  VectorType* new_Xi = new VectorType(Y[order + 1]);
157 
158  // now, produce the new representation
159  CurvePieceType* j0 = new CurvePieceType(this->t0(), Phi_j0, new_C_j0, m_r0, false, new_B_j0, new_invB_j0, Rem_j0, new_Xi, true);
160  addPiece(j0, true);
161  setValueAtCurrent(SetType(new_z, new_C_z, m_r0, new_B_z, new_invB_z, new_r));
162 
163  // this should never happen:
164  if (!(t_h == this->t0())) throw std::logic_error("DDESolutionCurve::move(): method has incompatible step size.");
165

194 template<typename SetSpec>
195 template<typename DynSysSpec>
196 void DDESolutionCurve<SetSpec>::epsilonShift(
197  DynSysSpec const& solver,
198  TimePointType const& at_t0, RealType const& epsilon,
199  DDESolutionCurve& out_result) const {
200  // this will test if the set is of good shape, more or less
201  try { out_result.set_r0(*(this->m_r0)); }
202  catch (std::logic_error &e) { throw rethrow("DDESolutionCurve::epsilonShift: bad r0 shape in result curve", e); }
203 
204  try {
205  // TODO: (FUTURE) this is only forward epsilon step. Implement backward step later
206  // TODO: (FUTURE) use solver and history to recompute Xi over shorter intervals t_i + [0, \epsi], when computing coefficients.
207  // TODO: (NOT URGENT) add checks for sanity of the input before proceeding.
208  TimePointType how_far = out_result.t0() - out_result.pastTime();
209  RealType HH = ecloseStep(RealType(m_grid(1)));
210  RealType EPSI = ecloseStep(epsilon);
211  auto ijet_out = out_result.begin();
212  auto ijet_this = this->at(at_t0 - how_far);
213  for (; ijet_out != out_result.end(); ++ijet_out){
214  (**ijet_this).evalAtDelta(epsilon, (**ijet_out)[0]);
215  size_type order = (**ijet_out).order();
216  for (size_type k = 1; k <= order; ++k){
217  (**ijet_this).evalCoeffAtDelta(k, epsilon, (**ijet_out)[k]);
218  }
219  VectorType xi1 = (**ijet_this).evalCoeffAtDelta(order + 1, HH);
220  ++ijet_this;
221  VectorType xi2 = (**ijet_this).evalCoeffAtDelta(order + 1, EPSI);
222  VectorType xi(xi1.dimension());
223  capd::vectalg::intervalHull(xi1, xi2, xi);
224  (**ijet_out).set_Xi(xi);
225  }
226  // we assume this extra jet is already computed,
227  // otherwise, we would need to call solver to extend solution - we do not want this.
228  (**ijet_this).evalAtDelta(epsilon, out_result.m_valueAtCurrent);
229  out_result.updateCommonR0();
230  } catch (std::logic_error &e) {
231  throw rethrow("DDESolutionCurve::epsilonShift: in shift algorithm algorithm", e);
232  }
233 }

235 template<typename SetSpec>
236 void DDESolutionCurve<SetSpec>::reduce(
237  DDESolutionCurve& out_result) const {
238  try {
239  if (out_result.length() != this->length()){
240  std::ostringstream info;
241  info << "DDESolutionCurve::reduce(): different length at output: " << out_result.length();
242  info << ", expected: " << this->length();
243  throw std::logic_error(info.str());
244  }
245  auto ijet_out = out_result.begin();
246  auto ijet_this = this->begin();
247  size_type jetIndex = 0;
248  RealType HH = ecloseStep(RealType(m_grid(1)));
249  for (; ijet_out != out_result.end(); ++ijet_out, ++ijet_this, ++jetIndex){
250  size_type order = (*ijet_out)->order();
251  if (order > (*ijet_this)->order()){
252  std::ostringstream info;
253  info << "DDESolutionCurve::reduce(): output order " << order;
254  info << " higher than input " << (*ijet_this)->order();
255  info << " at jet index " << jetIndex << ".";
256  throw std::logic_error(info.str());
257  }
258  auto icoeff_out = (*ijet_out)->beginJet();
259  auto icoeff_this = (*ijet_this)->beginJet();
260  for (; icoeff_out != (*ijet_out)->endJet(); ++icoeff_out, ++icoeff_this){
261  (*icoeff_out) = (*icoeff_this);
262  }
263  VectorType xi = (**ijet_this).evalCoeffAtDelta(order + 1, HH);
264  (**ijet_out).set_Xi(xi);
265  }
266  // we assume this extra jet is already computed,
267  // otherwise, we would need to call solver to extend solution - we do not want this.
268  out_result.m_valueAtCurrent = this->m_valueAtCurrent;
269  out_result.set_r0(this->get_r0());
270  out_result.updateCommonR0();
271  } catch (std::logic_error &e) {
272  throw rethrow("DDESolutionCurve::reduce(): in shift algorithm algorithm", e);
273  }
274 }
