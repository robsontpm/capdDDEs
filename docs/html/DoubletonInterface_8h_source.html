290  virtual Class& add(Class const & set){
291  size_type N0 = this->storageN0();
292  size_type d = this->dimension();
293  if (set.storageN0() != N0) throw std::logic_error("DoubletonInterface::add(set): Sets has incompatible N0 dimensions.");
294  if (set.dimension() != d) throw std::logic_error("DoubletonInterface::add(set): Sets has incompatible dimensions.");
295  VectorType other_r0 = set.get_r0();
296  VectorType this_r0 = this->get_r0();
297  if (!this->common_r0(&other_r0))
298  capd::vectalg::intervalHull(this_r0, other_r0, this_r0);
299  this->set_r0(this_r0);
300 
301  MatrixType new_C = this->get_C() + set.get_C(), S(d, N0);
302  capd::vectalg::split(new_C, S);
303  this->set_C(new_C);
304 
305  MatrixType invB = this->get_Binv();
306  VectorType new_x = get_x() + set.get_x(), s(d);
307  this->set_x(new_x);
308  capd::vectalg::split(new_x, s);
309  VectorType new_r = this->get_r() + (invB * set.get_B()) * set.get_r() + invB * (s + S * this_r0);
310  this->set_r(new_r);
311 
312  return *this;
313  }
315  virtual Class& mul(ScalarType const & c){
316  size_type d = this->dimension();
317  size_type N0 = this->storageN0();
318  MatrixType invB = this->get_Binv();
319  VectorType new_x = c * get_x(), s(d);
320  capd::vectalg::split(new_x, s);
321  MatrixType new_C = c * get_C(), S(d, N0);
322  capd::vectalg::split(new_C, S);
323  VectorType new_r = c * get_r() + invB * (S * this->get_r0() + s);
324  this->set_x(new_x);
325  this->set_C(new_C);
326  this->set_r(new_r);
327  // TODO: if c == 0. it would be good to reset B and invB for better performance later
328  return *this;
