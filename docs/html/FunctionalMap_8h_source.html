223  virtual void computeDDECoefficients(
224  const TimePointType& t0, const CurveType& x,
225  ValueStorageType& coeffs) const {
226  checkCurveDimension(x, "computeDDECoefficients");
227  VariableStorageType u; size_type order;
228  collectComputationData(t0, t0, RealType(0.), x, u, order);
229  if (coeffs.size() == 0 || order + 1 < coeffs.size())
230  coeffs.resize(order + 1);
231  ValueStorageType v; convert(u, v);
232  computeDDECoefficients(t0, v, coeffs);
233  }
