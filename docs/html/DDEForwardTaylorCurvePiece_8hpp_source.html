311 template<typename TimePointSpec, typename SetSpec, bool isInterval> // template spec
312 typename DDEForwardTaylorCurvePiece<TimePointSpec, SetSpec, isInterval>::VectorType // return type
313 DDEForwardTaylorCurvePiece<TimePointSpec, SetSpec, isInterval>::evalCoeffAtDelta(size_type n, const RealType& delta_t) const { // function decl
314  if (n == 0) return this->evalAtDelta(delta_t);
315  else if (n == m_order + 1) return *(this->m_Xi); // this is very rough, but we cannot produce better!
316  else if (n > m_order + 1) { // we cannot guarantee the continuity of higher class.
317  std::ostringstream info;
318  info << "DDEForwardTaylorCurvePiece::evalCoeff: cannot eval derivative of higher order. ";
319  info << "Max order available: " << m_order + 1 << ", requested order: " << n << ".";
320  throw std::logic_error(info.str());
321  }
322  // otherwise we need to do some computation...
323  VectorType result = *(this->m_Xi);
324  size_type j = this->m_order;
325  const_iterator it = this->backJet();
326  while (true){
327  result = VectorType(*it) + result * (delta_t * (RealType(j + 1) / RealType(j + 1 - n)));
328  if (j == n) break; else { --j; --it; } // makes sure that j is good if size_type is unsigned! (case n=0)
329  }
330  return result;
331 }

341 template<typename TimePointSpec, typename SetSpec, bool isInterval>
342 void DDEForwardTaylorCurvePiece<TimePointSpec, SetSpec, isInterval>::evalCoeffAtDelta(
343  size_type n, const RealType& delta_t, SetSpec& out) const {
344  if (out.dimension() != this->dimension())
345  throw std::logic_error("DDEForwardTaylorCurvePiece::evalCoeffAtDelta(): output has different dimension;");
346  if (out.storageN0() != this->storageN0())
347  throw std::logic_error("DDEForwardTaylorCurvePiece::evalCoeffAtDelta(): output has different N0 dimension;");
348  if (n == 0) { this->evalAtDelta(delta_t, out); return; }
349  else if (n == m_order + 1) { out.add(*(this->m_Xi)); return; }
350  else if (n > m_order + 1) {
351  std::ostringstream info;
352  info << "DDEForwardTaylorCurvePiece::evalCoeff: cannot eval derivative of higher order. ";
353  info << "Max order available: " << m_order + 1 << ", requested order: " << n << ".";
354  throw std::logic_error(info.str());
355  }
356  // Tried to do .mulThenAdd() to DRY but it produces undesirable effects than
357  // applying mean value form to the evaluation (the old way).
358  // mulThenAdd reorganizes the set at each step which might cause dependency error
359  // (when multipled again and again by delta_t)
360  size_type j = this->m_order;
361  size_type d = this->dimension();
362  size_type N0 = this->storageN0();
363  VectorType* x = new VectorType(d);
364  MatrixType* C = new MatrixType(d, N0);
365  // we will forget to reorganize B * r part - we evaluate it as Interval set here.
366  // it would be not feasible to maintain this structure in the algorithm.
367  (*x) += *(this->m_Xi);
368  const_iterator coeff = this->backJet();
369  while(true) {
370  (*x) = coeff->get_x() + coeff->get_B() * coeff->get_r() + (RealType(j + 1) / RealType(j + 1 - n)) * delta_t * (*x);
371  (*C) = coeff->get_C() + (RealType(j + 1) / RealType(j + 1 - n)) * delta_t * (*C);
372  if (j == n) break; else { --j; --coeff; }
373  }
374  MatrixType Id(d, d); Id.setToIdentity();
375  VectorType s(d); MatrixType S(d, N0);
376  capd::vectalg::split(*x, s);
377  capd::vectalg::split(*C, S);
378  s += S * (*m_r0);
379  out.set_x(x, true);
380  out.set_Cr0(C, m_r0, true, false);
381  out.set_B(Id);
382  out.set_r(s);
383 }

387 DDEForwardTaylorCurvePiece<TimePointSpec, SetSpec, isInterval>::taylorAtDelta(const RealType& delta_t) const { // function decl
388  VectorType result(dimension());
389  const_iterator coeff = endJet();
390  const_iterator stop = beginJet();

392  result = VectorType(*coeff) + delta_t * result;
393 
394  return result;
395 }

397 template<typename TimePointSpec, typename SetSpec, bool isInterval>
398 void DDEForwardTaylorCurvePiece<TimePointSpec, SetSpec, isInterval>::taylorAtDelta(
399  const RealType& delta_t, SetSpec& out) const {
400  const_iterator coeff = endJet();
401  const_iterator stop = beginJet();
402  out *= 0.0; // this will assure ownership... and zero the output (do I want this?)
403  size_type d = this->dimension();
404  size_type N0 = this->storageN0();
405  VectorType* x = new VectorType(d);
406  MatrixType* C = new MatrixType(d, N0);
407  while(coeff-- != stop){
408  (*x) = coeff->get_x() + coeff->get_B() * coeff->get_r() + delta_t * (*x);
409  (*C) = coeff->get_C() + delta_t * (*C);
410  }
411 
412  VectorType s(d); MatrixType S(d, N0);
413  MatrixType Id(d, d); Id.setToIdentity();
414  // we use Id as B part, as we do not want to decide how to choose B from the taylor mess.
415  capd::vectalg::split(*x, s);
416  capd::vectalg::split(*C, S);
417  s += S * (*m_r0);
418  out.set_x(x, true);
419  out.set_Cr0(C, m_r0, true, false); // TODO: (SERIOUS RETHINK): now the out and this curve piece shares r0, might be dangerous if this piece disappears (I had some issues with that)
420  out.set_B(Id);
421  out.set_Binv(Id);
422  out.set_r(s);
423 }

427 typename DDEForwardTaylorCurvePiece<TimePointSpec, SetSpec, isInterval>::VectorType // return type
428 DDEForwardTaylorCurvePiece<TimePointSpec, SetSpec, isInterval>::summaAtDelta(const RealType& delta_t) const { // function decl
429  // NOTE: (n+1) \int_0^t \xi(s) (t-s)^n ds with \xi(s) \in M - interval, translates
430  // into bound of type M * t^{n+1}, since
431  // (n+1) \int_0^t \xi(s) (t-s)^n ds \in M * (n+1) \int_0^t (t-s)^n ds = M * (n+1)/(n+1) (t-s)^{n+1}|_0^t
432  return (get_Xi() * power(delta_t, order() + 1));
433 }
