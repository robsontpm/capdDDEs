88  static bool increment(iterator from, iterator to){
89  iterator p = from;
90  while (p < to && p->isFinished()){

158  for (int i = 0; i < m_r0.dimension(); ++i){
159  auto r = m_r0[i].rightBound(); // we assume m_r0 is symmetric!
160  if (r == 0. && !done()){

162  }else{
163  auto diff = box[i] - m_x0[i];
164  // let diff = [left, right]
165  // we looking for i \in \Z such that -r -i * r < left,
166  // so we get i < d, with d = (left/r+1)/2, analogously
167  // for right we get i > d = (right/r-1)/2
168  // in case d \in \Z, we need to include extra piece, just in case
169  auto lod = (((diff / r) + 1) / 2.0).leftBound();
170  auto upd = (((diff / r) - 1) / 2.0).rightBound();

259  auto scan_for_token = [&](char expected){
