94  to_dat(out, typename CurveSpec::RealType(t0)); out << " "; to_dat(out, curve.eval(t0)); out << "\n";
95  StepSpec t = StepSpec(t0) + h;

98  to_dat(out, typename CurveSpec::RealType(t)); out << " "; to_dat(out, curve.eval(t)); out << "\n";

100  std::cout << "Draw Error at t = " << t << std::endl;
101  // do nothing...
102  }
103  t += h;
104  }
105  to_dat(out, typename CurveSpec::RealType(t1)); out << " "; to_dat(out, curve.eval(t1)); out << "\n";
106 }

324  auto dirandname = split_pathprefix(pathprefix);
325  auto dirpath = dirandname.first;
326  auto fileprefix = dirandname.second;
327  #ifdef DDES_ALLOW_SYSTEM
328  if (dirpath != "") capd::ddeshelper::runSystemCommand(std::string("mkdir -p " + dirpath));
329  #endif
330  // TODO: (NOT URGENT) we assume that curve.begin() has the lowest order
331  // TODO: (NOT URGENT) we also assume that all have the same order, so we plot Xi
332  typedef typename CurveSpec::size_type size_type;
333  typedef typename CurveSpec::VectorType VectorType;
334  typedef typename CurveSpec::RealType RealType;
335  size_type n = (*curve.begin())->order();
336  std::vector<std::string> dat_filenames;
337  for (size_type k = 0; k <= n; k++){
338  std::ostringstream filename;
339  filename << fileprefix << (fileprefix != "" ? "--" : "") << k << "-th--coeff.dat";
340  dat_filenames.push_back(filename.str());
341  std::ofstream out(dirpath + dat_filenames.back()); out.precision(15);
342  for (auto jet = curve.begin(); jet != curve.end(); ++jet){
343  VectorType v = (**jet)[k];
344  to_dat(out, RealType((*jet)->t0())); out << " "; to_dat(out, v); out << "\n";
345  }
346  out.close();
347  }
348  std::ostringstream filename;
349  filename << fileprefix << (fileprefix != "" ? "--" : "") << (n+1) << "-th--coeff.dat";
350  dat_filenames.push_back(filename.str());
351  std::ofstream out(dirpath + dat_filenames.back()); out.precision(15);
352  for (auto jet = curve.begin(); jet != curve.end(); ++jet){
353  VectorType v = (*jet)->get_Xi();
354  to_dat(out, RealType((*jet)->t0())); out << " "; to_dat(out, v); out << "\n";
355  }
356  out.close();
