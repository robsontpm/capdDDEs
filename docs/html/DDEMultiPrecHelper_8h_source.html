58 template<typename MatrixSpec>
59 void to_mpi_matrix(const MatrixSpec& inA, MpIMatrix& outA, int precission = 512){
60  int old_precision = MpReal::getDefaultPrecision();
61  MpReal::setDefaultPrecision(precission);
62  int size = inA.numberOfColumns(); // assume square matrix
63  outA = MpIMatrix(size, size);
64  for (int i = 0; i < size; ++i)
65  for (int j = 0; j < size; ++j)
66  outA[i][j] = MpInterval(inA[i][j].leftBound(), inA[i][j].rightBound());
67  MpReal::setDefaultPrecision(old_precision);
68 }
70 template<typename MatrixSpec>
71 void from_mpi_matrix(const MpIMatrix& inA, MatrixSpec& outA){
72  int size = inA.numberOfColumns(); // assume square matrix
73  outA = MatrixSpec(size, size);
74  for (int i = 0; i < size; ++i)
75  for (int j = 0; j < size; ++j)
76  outA[i][j] = mpi_to_interval(inA[i][j]);
77 }

98 template<typename MatrixSpec>
99 MpIMatrix mpi_inverse(MatrixSpec& A, int precission = 512){
100  int old_precision = MpReal::getDefaultPrecision();
101  MpReal::setDefaultPrecision(precission);
102  MpIMatrix mpA = double_to_mpi_matrix(A, precission);
103  mpA.Transpose();
104  MpIMatrix mpAinv = capd::matrixAlgorithms::inverseMatrix(mpA);
105  mpAinv.Transpose();
106  MpReal::setDefaultPrecision(old_precision);
107  return mpAinv;
108 }
