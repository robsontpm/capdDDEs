365  void integrateUntilSectionCrossing(CurveType& curve, RealType& timeBeforeSection, SingleStepFn stepFn){
366  ScalarType sign = m_section(curve);
367  // m_dynsys(curve);
368  (this->*stepFn)(curve);
369  sign = m_section(curve);
370  m_steps = 1;
371  try { checkSteps(); }
372  catch (std::logic_error& e) { throw rethrow("DDEBasicPoincareMap::approachSection: While leaving section: ", e); }
373 
374  //first we go to the right place - we want to leave section,
375  // or if we are on the opposite side, then go slightly after section, then proceed
376  while( (m_requiredSteps > 0 && m_steps < m_requiredSteps) || // we did not do enough steps
377  (sign == 0.0) || // We are on the section
378  !( // or section sign is not correct
379  (m_direction == CrossingDirection::Both)
380  || ((m_direction == CrossingDirection::MinusPlus) && (sign < 0.0))
381  || ((m_direction == CrossingDirection::PlusMinus) && (sign > 0.0))
382  )
383  ){
384  timeBeforeSection = curve.t0();
385  // m_dynsys(curve);
386  (this->*stepFn)(curve);
387  sign = m_section(curve);
388  ++m_steps;
389  try { checkSteps(); }
390  catch (std::logic_error& e) { throw rethrow("DDEBasicPoincareMap::approachSection: While leaving section: ", e); }
391  }
392  //do the iteration until section is crossed (we have already done enough steps)
393  while(m_direction * sign > 0.0){
394  timeBeforeSection = curve.t0();
395  // m_dynsys(curve);
396  (this->*stepFn)(curve);
397  sign = m_section(curve);
398  ++m_steps;
399  try { checkSteps(); }
400  catch (std::logic_error& e) { throw rethrow("DDEBasicPoincareMap::approachSection: While leaving section: ", e); }
401  }
402  }

478  void extractVariationalMatrix(
479  CurveType const& curve,
480  MatrixType& fullV, MatrixType& reducedV,
481  std::vector<size_type> reducedShape, int p_howFar = -1) const {
482  JacobianStorageType V, R; // it will be easier to iterate over the complete structure to join it into V matrix later.
483  bool doReduced = reducedShape.size() != 0;
484  size_type howFar = (p_howFar >= 0 ? p_howFar : curve.rend() - curve.rbegin());
485 
486  V.push_back(curve.getValueAtCurrent().getMatrixData());
487  if (doReduced) R.push_back(curve.getValueAtCurrent().getMatrixData());
488 
489  size_type pMax = reducedShape.size();
490  size_type pCount = 0;
491  for (auto ijet = curve.rbegin(); (ijet != curve.rend()) && (pCount < howFar); ++ijet, ++pCount){
492  size_type nCount = 0;
493  for (auto icoeff = (*ijet)->begin(); icoeff != (*ijet)->end(); ++icoeff, ++nCount){
494  V.push_back(icoeff->getMatrixData());
495  if (doReduced && pCount < pMax && nCount < reducedShape[pCount])
496  R.push_back(icoeff->getMatrixData());
497  }
498  }
499 
500  size_type allRowsCount = V.size() * curve.dimension();
501  size_type redRowsCount = R.size() * curve.dimension();
502  size_type columnsCount = V[0].size() * curve.dimension();
503 
504  size_type I = 0, J = 0;
505  size_type i = 0, j = 0;
506  fullV = MatrixType(allRowsCount, columnsCount);
507  for (auto a = V.begin(); a != V.end(); ++a){
508  J = 0;
509  for (auto b = a->begin(); b != a->end(); ++b){
510  for (i = 0; i < b->numberOfRows(); ++i)
511  for (j = 0; j < b->numberOfColumns(); ++j)
512  fullV[I+i][J+j] = (*b)[i][j];
513  J += j;
514  }
515  I += i;
516  }
517 
518  if (redRowsCount){
519  if (&fullV == &reducedV)
520  throw std::logic_error("Same storage variable used for full and reduced V. Might result is a crash.");
521  reducedV = MatrixType(redRowsCount, columnsCount);
522  size_type I = 0, J = 0;
523  size_type i = 0, j = 0;
524  for (auto a = R.begin(); a != R.end(); ++a){
525  J = 0;
526  for (auto b = a->begin(); b != a->end(); ++b){
527  for (i = 0; i < b->numberOfRows(); ++i)
528  for (j = 0; j < b->numberOfColumns(); ++j)
529  reducedV[I+i][J+j] = (*b)[i][j];
530  J += j;
531  }
532  I += i;
533  }
534  }
