204  void computePoincareMap(CurveType& curve, RealType& out_approachTime, CurveType& on_section, size_type n) {
205  // TODO: (!!!URGENT, RETHINK) check if the section is not crossed between steps. For this I need to have a method to obtain enclosures over intervals of the curve and this should take section into consideration.
206  // TODO: (!!!URGENT, RETHINK) My first bet is to put code in Section and require from Solution to be able to produceenclosures (after the step is made - becouse then I can simply produce estimate over current time)
207  // TODO: (!!!URGENT, RETHINK) this would go nicely with the imporvement to hold enclosures already computed in the representation to make better estimates.
208  bool isExtraStepNeeded = false;
209  RealType epsiLo, epsiUp;
210  integrateUntilSectionCrossing(curve, m_lastTimeBeforeSection, isExtraStepNeeded, n);
211  findCrossingTime(curve, on_section, epsiLo, -1);
212 
213  // TODO: (NOT URGENT): for some reason intervalHull from capd does not work on default capd::interval (using filib), therefore I have decided to write it myself; if capd repaired - you can chenge this to capd::intervals::intervalHull()
214  auto myIntervalHull = [](RealType& a, RealType b) {
215  auto lo = min(a.leftBound(), b.leftBound());
216  auto up = max(a.rightBound(), b.rightBound());
217  return RealType(lo, up);
218  };
219 
220  if (isExtraStepNeeded){
221  // TODO: (NOT URGENT) STRASZNIE SKOMPLIKOWANE ODTAD - Refactor later
222  m_lastEpsilonTime = epsiLo.leftBound();
223  RealType step = curve.getStep();
224  RealType fullStep = RealType(0.0, 1.0) * step;
225  epsiLo = RealType(epsiLo.leftBound(), RealType(curve.getStep()).rightBound());
226  curve.epsilonShift(m_dynsys, epsiLo, on_section);
227  CurveType extraImage = on_section;
228  bool done = false;
229  while (!done){
230  extraImage *= 0.;
231  curve.move(m_dynsys);
232  RealType sign = m_section(curve);
233  if (m_direction * sign < 0.){
234  done = true;
235  findCrossingTime(curve, extraImage, epsiUp, +1);
236  epsiUp = RealType(0., epsiUp.rightBound());
237  curve.epsilonShift(m_dynsys, epsiUp, extraImage);
238  }else{
239  curve.epsilonShift(m_dynsys, fullStep, extraImage);
240  m_lastEpsilonTime += fullStep;
241  }
242  // no matter if we did intermediate full step, or if we done last (up) step
243  // we need to hull intermediate result with what we have from lo.
244  capd::ddes::hull(on_section, extraImage, on_section);
245  }
246  //m_lastEpsilonTime += capd::intervals::intervalHull(epsiLo, epsiUp); // TODO: (check with Daniel) filib interval does not have this???
247  m_lastEpsilonTime += myIntervalHull(epsiLo, epsiUp);
248  // TODO: (NOT URGENT) DOTAD REFACTOR
249  }else{
250  findCrossingTime(curve, on_section, epsiUp, +1);
251  // m_lastEpsilonTime = capd::intervals::intervalHull(epsiLo, epsiUp); // TODO: (check with Daniel) as before
252  m_lastEpsilonTime = myIntervalHull(epsiLo, epsiUp);
253  curve.epsilonShift(m_dynsys, m_lastEpsilonTime, on_section);
254  }
255  out_approachTime = m_lastTimeBeforeSection + m_lastEpsilonTime;
256  }

278  void integrateUntilSectionCrossing(CurveType& curve, RealType& timeBeforeSection, bool& isExtraStepNeeded, size_type n=1){
279  ScalarType sign = m_section(curve);
280  curve.move(m_dynsys);
281  sign = m_section(curve);
282  //std::cout << "at t = " << ScalarType(curve.t0()) << " step: " << int(curve.t0()) << " sign: " << sign << std::endl;
283  m_steps = 1;
284  try { checkSteps(); }
285  catch (std::logic_error& e) { throw rethrow("DDEPoincareMap::approachSection: While initial leaving section: ", e); }
286 
287  for (size_type count = 0; count < n; ++count){
288  //first we go to the right place - we want to leave section,
289  // or if we are on the opposite side, then go slightly after section, then proceed
290  while( (m_requiredSteps > 0 && m_steps < m_requiredSteps) || // we did not do enough steps
291  (sign.contains(0.0)) || // We are on the section
292  !( // or section sign is not correct
293  (m_direction == CrossingDirection::Both)
294  || ((m_direction == CrossingDirection::MinusPlus) && (sign < 0.0))
295  || ((m_direction == CrossingDirection::PlusMinus) && (sign > 0.0))
296  )
297  ){
298  timeBeforeSection = curve.t0();
299  curve.move(m_dynsys);
300  sign = m_section(curve);
301  //std::cout << "at t = " << ScalarType(curve.t0()) << " step: " << int(curve.t0()) << " sign: " << sign << std::endl;
302  ++m_steps;
303  try { checkSteps(); }
304  catch (std::logic_error& e) { throw rethrow("DDEPoincareMap::approachSection: While approaching section: ", e); }
305  }
306  //std::cout << "Section approached at " << m_steps << " with sign: " << sign << " direction is " << m_direction << std::endl;
307  //do the iteration until section is crossed (we have already done enough steps)
308  while(m_direction * sign > 0.0){
309  timeBeforeSection = curve.t0();
310  curve.move(m_dynsys);
311  sign = m_section(curve);
312  //std::cout << "at t = " << ScalarType(curve.t0()) << " step: " << int(curve.t0()) << " sign: " << sign << std::endl;
313  ++m_steps;
314  try { checkSteps(); }
315  catch (std::logic_error& e) { throw rethrow("DDEPoincareMap::approachSection: While crossing section after approach: ", e); }
316  }
317  isExtraStepNeeded = sign.contains(0.);
318  }
319  }

325  void findCrossingTime(CurveType const& curve, CurveType& requested, RealType &crossingTime, int testDirection){
327  crossingTime = RealType(0.0, RealType(curve.getStep()).rightBound());
328  RealType epsi;
329  while (crossingTime.rightBound() - crossingTime.leftBound() > m_binsearchEpsilon){
330  requested *= 0;
331  epsi = 0.5 * (crossingTime.leftBound() + crossingTime.rightBound());
332  curve.epsilonShift(m_dynsys, epsi, requested);
333  // TODO: (NOT URGENT) this is so bad looking...
334  if (m_section(requested) * m_direction * testDirection < 0.0)
335  // I am on the good side of the section after step epsi (good - I wanted to be on this side)
336  crossingTime = testDirection < 0 ? RealType(epsi.leftBound(), crossingTime.rightBound()) : RealType(crossingTime.leftBound(), epsi.rightBound());
337  else
338  // I am on section or on the bad side of it
339  crossingTime = testDirection < 0 ? RealType(crossingTime.leftBound(), epsi.rightBound()) : RealType(epsi.leftBound(), crossingTime.rightBound());
340  }
341  }
