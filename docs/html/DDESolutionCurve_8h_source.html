276  size_type N0 = 0):

278  m_r0(new VectorType(N0)),

288  throw std::logic_error(info.str());

290  try {
291  for (TimePointType t = t0; t < t1; t += grid.point(1)){
292  FJet tt(f.imageDimension(), 1, order + 2);

294  *(tti) = ScalarType(t);
295  *(++tti) = 1.;

338  DDESolutionCurve subcurve(size_type index1, size_type index2) const {
339  TimePointType from = (index1 == m_pieces.size() ? t0() : m_pieces[index1]->t0());
340  DDESolutionCurve result(m_grid, from, dimension(), storageN0());
341  result.set_r0(get_r0());
342  for (size_type i = index1; i < index2; ++i){
343  result.addPiece(*m_pieces[i]);
344  }
345  result.setValueAtCurrent(index2 == m_pieces.size()? this->m_valueAtCurrent : (*m_pieces[index2])[0]);
346  return result;
