162  out_encl.clear();
163  size_type d = this->imageDimension();
164  // we collect jets and compute admissible order
165  // TODO: (NOT URGENT) rewrite using pointers to Jets (will be faster)!
166  out_admissible_order = 40;
167  std::vector<JetType> jets;
168  for (auto tau = begin(); tau != end(); ++tau){
169  jets.push_back(x.jet(t0 - *tau));
170  size_type k = jets.back().order();
171  if (k + 1 < out_admissible_order) out_admissible_order = k + 1;
172  }
173  // Then we fill in at u[0] the value at current time (always present in u, by assumption)
174  out_u.push_back(x.getValueAtCurrent());
175  out_encl.push_back(VectorType(d)); // just make room for future.
176  // and then all jets, order by order (in this way it is easier to fill AutoDiff later)
177  ValueStorageType pastData;
178  for (auto jit = jets.begin(); jit != jets.end(); ++jit){
179  out_u.push_back((*jit)[0]);
180  VectorType enc = jit->evalAtDelta(dt);
181  out_encl.push_back(enc);
182  pastData.push_back(enc);
183  }
184  for (size_type k = 1; k < out_admissible_order; ++k){
185  for (auto jit = jets.begin(); jit != jets.end(); ++jit){
186  out_u.push_back((*jit)[k]);
187  out_encl.push_back(jit->evalCoeffAtDelta(k, dt));
188  }
189  }
190  // add extra data necessary in current Solvers, that is, global estimate of one higher order.
191  for (auto ijet = jets.begin(); ijet != jets.end(); ++ijet)
192  out_encl.push_back(ijet->evalCoeffAtDelta(out_admissible_order, dt));
193 
194  // run find enclosure, we have everything we need.
195  VectorType dump_Z(d); // it is not used
196  // in a rare event, that we are computing for 0 length interval...
197  if (t0 == th){
198  out_encl[0] = out_u[0];
199  }else{
200  findRoughEnclosure(t0, dt, pastData, out_u[0], dump_Z, out_encl[0]);
201  }

242  virtual void computeDDECoefficients(
243  const RealType& t0, const ValueStorageType& u, // input
244  ValueStorageType& coeffs) const { // output
245  // we assume u[0] is always present, and all other u[i] has the same
246  checkDimension(u[0]);
247  // we assume that coeffs has a proper size for the computation.
248  // for example, set when user took data u with collectComputationData()
249  size_type resultOrder = coeffs.size() - 1;
250 
251  typedef fadbad::T<ScalarType> AutoDiffTScalar; // Auto Diff T (Taylor) scalar type
252  typedef typename VectorType::template rebind<AutoDiffTScalar>::other TADVectorType;
253  TADVectorType args(dimension());
254  // propagate arguments in a given order - first value at t0
255  // then jet of \tau_1, then \tau_2, etc...
256  // this is change from the initial paper!, where I have done delay terms first
257  // this is coherent with what is in the general DDE literature though...
258  // we assume this order in u (because we produce itd this way in collectComputationData()!).
259  auto current_u = u.begin();
260  auto iarg = args.begin();
261  for (size_type idelay = 0; idelay <= delaysCount(); ++idelay, ++current_u)
262  for (auto iu = current_u->begin(); iu != current_u->end(); ++iu, ++iarg)
263  (*iarg)[0] = *iu;
264 
265  // we are ready to go for the recurrent relation on coefficients.
266  coeffs[0] = u[0]; // by definition, see paper
267  for (size_type k = 1 ; k <= resultOrder; ++k){
268  // eval Autodiff routine so that we would be able to extract v[k-1] = F^{[k-1]}(...)
269  TADVectorType v(imageDimension());
270  m_map(t0, args, v);
271  // propagate AD jets and output.
272  // this will be F_k from the latest paper on DDE integration.
273  VectorType Fk(imageDimension());
274  auto iarg = args.begin();
275  for (size_type i = 0; i < imageDimension(); ++i, ++iarg){
276  v[i].eval(k-1);
277  Fk[i] = v[i][k-1];
278  // propagate computed recurrent formula into args with proper scaling!
279  (*iarg)[k] = Fk[i] / ScalarType(k);
280  }
281  // this is the right formula from the last paper!
282  coeffs[k] = Fk / ScalarType(k);
283  // do not propagate anything else if we are past the requestedOrder (it might cause errors)
284  if (k == resultOrder) break;
285  // propagate higher order coefficients in the past to the args Vector.
286  // the recurrent Taylor formula is already there.
287  for (size_type idelay = 1; idelay <= delaysCount(); ++idelay, ++current_u){
288  for (auto iu = current_u->begin(); iu != current_u->end(); ++iu, ++iarg)
289  (*iarg)[k] = *iu;
290  }
291  }
292  }

297  virtual void computeDDECoefficients(
298  const RealType& t0, const ValueStorageType& u, // input
299  ValueStorageType& coeffs, JacobianStorageType& Du) const { // output
300  // we assume u[0] is always present, and all other u[i] has the same
301  checkDimension(u[0]);
302  // we assume that coeffs has a proper size for the computation.
303  // for example, set when user took data u with collectComputationData()
304  size_type resultOrder = coeffs.size() - 1;
305  // d = space dimension of a single variable in the map / input data in ValueStorageType (must be the same).
306  size_type d = this->imageDimension();
307  size_type numDelays = delaysCount();
308  // those are common matrices of dimension M(d, d),
309  MatrixType Id(d, d); Id.setToIdentity();
310  MatrixType Zero(d, d);
311 
312  // This is obvious
313  size_type partialsCount = u.size();
314  // in partialsDimension we keep how many actual coefficients from all of the Jets used to
315  // compute rhs of the equation. As we are computing jets up to coeffs.order() (incl.).
316  // then we are going to use only coefficients of all jets up to coeffs.order() - 1 (incl).
317  // Thus, the formula. Please note in all those computations we have 0,..,resultOrder-1
318  // Taylor coefficients in any jet used. Therefore the number of Taylor coeffs at each
319  // jet used is resultOrder. This is trivial, but I want this to be reiterated for readability.
320  size_type partialsDimension = d * u.size();
321  // we make room to store Jacobian matrices. Each matrix will be M(d,d).
322  // Du is shorthand for $D_u coeffs$
323  // in Du[i][j] we will have \partial coeffs[i]
324  Du.clear(); Du.resize(resultOrder + 1);
325  for (auto iDu = Du.begin(); iDu != Du.end(); ++iDu)
326  iDu->resize(partialsCount);
327 
328  typedef fadbad::F<ScalarType> FADScalar; // Auto Diff Forward scalar type (i.e. w.r.t. initial coefficients)
329  typedef fadbad::T<FADScalar> TFADScalar; // Auto Diff Taylor type with option to compute derivatives wrt initials
330  typedef typename VectorType::template rebind<TFADScalar>::other TFADVectorType;
331  // this is Automatic Differentiation both w.r.t t but also w.r.t. coefficients.
332  // we will use it to automatically eval the r.h.s. of the equation with all
333  // partial derivatives when recursively compute Taylor coefficients at current t0
334  TFADVectorType args(dimension());
335 
336  // we will successively fill-in args as the recursion will go on
337  auto current_u = u.begin();
338  auto iarg = args.begin();
339  // this will hold current number of parameter (among 0,.., partialsDimension-1)
340  // we need those for AutoDiff types (they use this kind of numbering)
341  size_type partialAD = 0;
342  // this will hold current number of variable in u (among 0,..,partialsCount-1, each variable is d dimensional)
343  // we need those for our purpose (to match scalar variables to appropriate entries in vector/matrices)
344  size_type partialu = 0;
345  // we fill-in initial values for the recurrence, i.e. the value at t0 stored
346  // in u[0] and values at values in the past, at given delays.
347  // We mark each of those values as dependent only on self. This will give us
348  // possibility to automatically compute dependence of the formulas on them
349  // i.e. the value \frac{\partial coeffs}{\partial u[partialu]}
350  // Note: here k = 0, and we iterate idelay from 0. Later, it will be from 1. See docs there.
351  for (size_type idelay = 0; idelay <= numDelays; ++idelay, ++current_u, ++partialu){
352  for (auto iu = current_u->begin(); iu != current_u->end(); ++iarg, ++iu, ++partialAD){
353  // (*iarg)[0] is [0]-th order Taylor coefficient
354  (*iarg)[0] = *iu;
355  // and we mark it as dependent only on self among all 0,..,partialsDimension-1 variables
356  // (i.e. \frac{\partial{(*iarg)[0]}}{\partial{(*iarg)[0]}} = 1, and 0 otherwise)
357  (*iarg)[0].diff(partialAD, partialsDimension);
358  }
359  }
360 
361  // we compute coeffs[0] and derivatives...
362  coeffs[0] = u[0]; // coeffs[0] = x(t0) by definition of (see paper)
363  Du[0][0] = Id; // \frac{Dcoeffs_{[0]}}{Dx(t0)} = Id (as a matrix M(d, d)) (since coeffs[0] = x(t0))
364  // Next... \frac{Dcoeffs_{[0]}}{{Djet_m}_{[mk]}} = 0 (as a matrix M(d, d)) (since coeffs[0] = x(t0))
365  for (size_type par = 1; par < partialsCount; ++par)
366  Du[0][par] = Zero;
367 
368  // then we compute higher order coeffs, with recurrent Taylor formula (see paper)
369  for (size_type k = 1; k <= resultOrder; ++k){
370  // eval AutoDiff routine so we would be able to extract v[.][k-1] = F^{[k-1]}(...)
371  // together with partial derivatives with respect to coefficients in the past.
372  TFADVectorType v(imageDimension());
373  m_map(t0, args, v);
374 
375  // propagate AD jets and output.
376  VectorType Fk(imageDimension());
377  auto iarg = args.begin();
378  for (size_type i = 0; i < imageDimension(); ++i, ++iarg){
379  v[i].eval(k-1);
380  Fk[i] = v[i][k-1].val(); // here we need to .value(), to convert from FAD to VectorType
381  (*iarg)[k] = v[i][k-1] / ScalarType(k); // here we need to propagate things with all partial derivatives! So we copy from v (appropriate AutoDiff Type)
382  }
383 
384  // we output the value, properly scaled
385  coeffs[k] = Fk / ScalarType(k);
386  // and partial derivatives w.r.t all previous coeffs.
387  MatrixType parDu(d, d);
388  for (size_type par = 0; par < partialu; ++par){
389  // we copy the appropriate partial derivatives into a local Matrix, which will be later
390  // Du[k][par] = \frac{\partial coeffs[k]}{\partial u[par]}
391  for (size_type i = 0; i < d; ++i) // we use formula par*d+j
392  for (size_type j = 0; j < d; ++j) // to obtain the local index
393  parDu[i][j] = v[i][k-1].d(par * d + j) / ScalarType(k); // among all partialsCount variables
394  Du[k][par] = parDu;
395  }
396  // mark that this order coeff does not depend on higher order u's
397  for (size_type par = partialu; par < partialsCount; ++par)
398  Du[k][par] = Zero;
399 
400 
401  // this if is to ensure that we do not copy, if there is no more values left in u.
402  if (k == resultOrder) break;
403 
404  // We have now propagated dependence of the solution on the value at
405  // present time t0 to Du. Now, finally, we propagate higher order
406  // coefficients in the past to args for the next level of recurrence.
407  for (size_type idelay = 1; idelay <= numDelays; ++idelay, ++current_u, ++partialu){
408  for (auto iu = current_u->begin(); iu != current_u->end(); ++iarg, ++iu, ++partialAD){
409  // (*iarg)[0] is [0]-th order Taylor coefficient
410  (*iarg)[k] = *iu;
411  // and we mark it as dependent only on self among all 0,..,partialsDimension-1 variables
412  // (i.e. \frac{\partial{(*iarg)[0]}}{\partial{(*iarg)[0]}} = 1, and 0 otherwise)
413  (*iarg)[k].diff(partialAD, partialsDimension);
414  }
415  }
416  // TODO: (FUTURE) rewrite on iterators only, it is possible for sure!
417  // TODO: (FUTURE) in such a case probably, I would not need many of the *Count variables.
418 
419  // now we are ready for the next level of recursion.
420  } // for k loop
421  } // function computeDDECoefficients

471  void findRoughEnclosure(
472  RealType const & t0, RealType const & dt, // input
473  ValueStorageType const & pastData, VectorType const & z, // input
474  VectorType& out_Z, VectorType& out_Y) const { // output
475 
476  // some helper constants
477  size_type d = this->imageDimension();
478  RealType trialStep = ScalarType(-ROUGH_TRIALSTEP, 1.0 + ROUGH_TRIALSTEP) * dt;
479  RealType h = RealType(0.0, 1.0) * dt;
480  VectorType Small(d);
481  for (size_type i = 0; i < d; ++i)
482  Small[i] = ScalarType(-1.0, 1.0) * 1e-21;
483 
484  // we will use x to evaluate the function F on it
485  VectorType x(this->dimension()); auto ix = x.begin();
486  // we will need to update current time t0 data each time later...
487  for (auto ival = z.begin(); ival != z.end(); ++ival, ++ix)
488  (*ix) = (*ival);
489  // but we only need to copy past data once.
490  for (auto ipast = pastData.begin(); ipast != pastData.end(); ++ipast)
491  for (auto ival = ipast->begin(); ival != ipast->end(); ++ival, ++ix)
492  (*ix) = (*ival);
493 
494  VectorType Fx(d);
495  m_map(t0 + h, x, Fx); // eval F(x) to Fx
496  out_Z = z + trialStep * Fx + Small; // initial very rough guess
497 
498  bool found = false;
499  size_type counter = 0; // counts until max number of heuristic trials is done (see ROUGH_LIMIT)
500  while ((!found) && (counter++ < ROUGH_LIMIT)){
501  // copy Z as the first variable in x
502  auto ix = x.begin();
503  for (auto ival = out_Z.begin(); ival != out_Z.end(); ++ival, ++ix)
504  (*ix) = (*ival);
505  m_map(t0 + h, x, Fx);
506  out_Y = z + h * Fx;
507  found = true;
508  for (size_type i = 0; i < d; ++i){
509  if(!(out_Y[i].subsetInterior(out_Z[i]))){
510  found = false;
511  // Z to small in the i-th coordinate, grow Y and use as Z for another try
512  out_Z[i] = out_Y[i];
513  ScalarType s;
514  out_Z[i].split(s);
515  s = ROUGH_MULFACT * s;
516  out_Z[i] += s;
517  }
518  }
519  }
520 
521  if(!found)
522  throw std::logic_error("DiscreteDelaysFunctionalMap::findRoughEnclosure(): Rough enclosure not found!");
523 
524  // try to refine estimates
525  VectorType save_Y(d);
526  VectorType save_Z(d);
527 
528  // first (new) strategy to refine
529  found = true;
530  counter = 0;
531  do{
532  save_Y = out_Y;
533  save_Z = out_Z;
534  out_Z = out_Y; // Try current Y as Z and compute new Y
535  // this is very old version
536  auto ix = x.begin();
537  for (auto ival = out_Z.begin(); ival != out_Z.end(); ++ival, ++ix)
538  (*ix) = (*ival);
539  m_map(t0 + h, x, Fx);
