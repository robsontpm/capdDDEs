54 template<typename SetSpec>
55 void hull(DDESolutionCurve<SetSpec> const& X, DDESolutionCurve<SetSpec> const& Y, DDESolutionCurve<SetSpec>& Z){
56  if (X.length() != Y.length())
57  throw std::logic_error("capd::ddes::hull(): X and Y curves have different lengths.");
58  if (X.length() != Z.length())
59  throw std::logic_error("capd::ddes::hull(): output curve has different length.");
60  if (X.dimension() != Y.dimension())
61  throw std::logic_error("capd::ddes::hull(): X and Y have different dimensions!");
62  if (X.dimension() != Z.dimension())
63  throw std::logic_error("capd::ddes::hull(): output storage has different dimensions!");
64  if (X.storageN0() != Y.storageN0())
65  throw std::logic_error("capd::ddes::hull(): X and Y have different N0 dimensions!");
66  if (X.storageN0() != Z.storageN0())
67  throw std::logic_error("capd::ddes::hull(): output storage has different N0 dimensions!");
68  if (X.storageDimension() != Y.storageDimension())
69  throw std::logic_error("capd::ddes::hull(): X and Y have different storage dimensions!");
70  if (X.storageDimension() != Z.storageDimension())
71  throw std::logic_error("capd::ddes::hull(): output storage has different storage dimensions!");
72 
73  // we do not care about different time points at Jets.
74  auto xit = X.begin();
75  auto yit = Y.begin();
76  auto zit = Z.begin();
77  for ( ; xit != X.end(); ++xit, ++yit, ++zit){
78  if ((*xit)->order() != (*yit)->order())
79  throw std::logic_error("capd::ddes::hull(): X and Y have different order at some point");
80  if ((*xit)->order() != (*zit)->order())
81  throw std::logic_error("capd::ddes::hull(): output storage has different order at some point");
82  auto xcend = (*xit)->endJet();
83  auto xcoeff = (*xit)->beginJet();
84  auto ycoeff = (*yit)->beginJet();
85  auto zcoeff = (*zit)->beginJet();
86  for ( ; xcoeff != xcend; ++xcoeff, ++ycoeff, ++zcoeff)
87  capd::ddes::hull(*xcoeff, *ycoeff, *zcoeff);
88  }
89  capd::ddes::hull(X.getValueAtCurrent(), Y.getValueAtCurrent(), Z.getValueAtCurrent());
90 }
