172  Solution integrate(int iters, const Vector& initial, Vector& result, bool use_extension=false){
173  if (initial.dimension() != DIMENSION && initial.dimension() != M()){
174  throw std::logic_error("NonrigorousHelper::initialIteration(): vector must be either DIMENSION (for const) or M dimensional (for any initial)");
175  }
176  if (iters < 0) iters = -m_p * iters;
177  auto tau = m_grid.point(m_p);
178  auto t_0 = m_grid.point(0);
179  Vector v(DIMENSION); if (initial.dimension() == DIMENSION) v = initial;
180  Solution X(m_grid, -tau, t_0, m_n, v);
181  Solution XXX(m_grid, -tau, t_0, m_n, v);
182  if (initial.dimension() == X.storageDimension()){ X.set_x(initial); XXX.set_x(initial); }
183  DDEq dde(Eq(m_params), tau);
184  Solver solver(dde, use_extension ? m_maxOrder : m_n);
185  for (int i = 0; i < iters; i++)
186  solver(XXX);
187  result = XXX.subcurve(XXX.currentTime() - tau).get_x();
188  return XXX;
189  }

221  Real iteratePoincare(std::ostream& info, int iters, JacJetSection& section, Vector &x, Matrix& V, Matrix& DP){
222  info << "# iteratePoincare START" << std::endl;
223  auto tau = m_grid.point(m_p);
224  auto t_0 = m_grid.point(0);
225  JacSolution X(m_grid, -tau, t_0, m_n, Vector(DIMENSION));
226  X.set_x(x);
227 
228  JacDDEq dde(Eq(m_params), tau);
229  JacSolver solver(dde, m_maxOrder);
230  JacPoincareMap pm(solver, section);
231  pm.setDirection(this->crossingDirection);
232  pm.setRequiredSteps(m_reqSteps);
233  pm.setMaxSteps(m_maxSteps);
234 
235  Real reachTime; Jacobians Jac;
236  capd::vectalg::EuclLNorm<Vector, Matrix> euclNorm;
237  JacSolution on_section(X); on_section *= 0.;
238  pm(X, on_section, reachTime);
239  X = on_section; on_section *= 0.;
240  info << "# time of X on section: " << X.getT0() << " should be 0" << std::endl;
241  info << "# reqSteps: " << pm.getRequiredSteps() << std::endl;
242  info << "# maxSteps: " << pm.getMaximumSteps() << std::endl;
243  double diff;
244  for (int i = 0; i < iters; i++){
245  Vector xx = X.get_x();
246  pm(X, on_section, reachTime);
247  Vector yy = on_section.get_x();
248  diff = euclNorm(xx-yy);
249  X = on_section; on_section *= 0.;
250  info << "# reqSteps: " << pm.getRequiredSteps() << std::endl;
251  info << "# maxSteps: " << pm.getMaximumSteps() << std::endl;
252  info << "# time of X on section: " << X.getT0() << " should be 0" << std::endl;
253  info << "# reachTime = " << reachTime << std::endl;
254  info << "# ||PX - X|| = " << diff << std::endl;
255  }
256  info << "# Final poincare: " << std::flush;
257  pm.setRequiredSteps(m_reqSteps); info << "#" << std::flush;
258  pm.setMaxSteps(m_maxSteps); info << "#" << std::flush;
259  x = X.get_x(); info << "#" << std::flush;
260  Vector Px, fPx; info << "#" << std::flush;
261  pm.setInitialV(X); info << "#" << std::flush;
262  JacSolution PX(X); PX *= 0.; info << "#" << std::flush;
263  pm(X, PX, reachTime, x, Px, fPx, V, DP); info << "#" << std::flush;
264  diff = euclNorm(x - PX.get_x()); info << " DONE" << std::endl;
265  info << "# iteratePoincare END" << std::endl;
266  return diff;
267  }

298  info << "refinePeriodic START" << std::endl;

307  JacPoincareMap pm(solver, section);

312  capd::vectalg::EuclLNorm<Vector, Matrix> euclNorm;
313  Real reachTime;
314  Vector Px, fPx, dx(X.storageDimension());
315  Matrix Id(X.storageDimension(), X.storageDimension()); Id.setToIdentity();
316  pm.setInitialV(X);

318  pm(X, PX, reachTime, x, Px, fPx, V, DP);
319  double diff = euclNorm(Px-x);
320  info << "# reachTime = " << reachTime << std::endl;
321  info << "# ||PX - X|| = " << diff << std::endl;
322  info << "# Solving for Newton..." << std::flush;
323  capd::matrixAlgorithms::gauss(DP-Id, x-Px, dx);
324  info << "# DONE!" << std::endl;
325  x += dx;
326  info << "# refinePeriodic END" << std::endl;
327  return diff;

596  void rawLoadSetup(std::istream& in, size_type &p, size_type &n, Vector &params);
597  void updateSteps(bool control_steps = true) {
598  m_reqSteps = (m_reqSteps < 0 ? -m_reqSteps * m_p : m_reqSteps);
599  if (control_steps)
600  m_reqSteps = (m_reqSteps < m_p ? m_p : m_reqSteps);
