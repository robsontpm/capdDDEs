397  Solution integrate(int iters, const Solution& initial, Solution& result, bool use_extension=false){
398  try {
399  checkGrid(initial); checkGrid(result);
400  } catch (std::logic_error& e){
401  throw capd::ddes::rethrow("NonrigorousHelper::integrate(int, Solution, Solution, bool): initial and result must have the same grid!", e);
402  }
403  if (iters < 0) iters = -m_p * iters;
404  auto tau = m_grid.point(m_p);
405  DDEq dde(Eq(m_params), tau);
406  Solver solver(dde, use_extension ? m_maxOrder : m_n);
407  Solution XXX = initial;
408  for (int i = 0; i < iters; i++)
409  solver(XXX);
410  Solution X = XXX.subcurve(XXX.currentTime() - tau);
411  capd::ddeshelper::copyReduce(X, result);
412  return XXX;
413  }

727  Real refinePeriodic(std::ostream& info, C1JetSection& section, Vector& x, Matrix& V, Matrix&DP){
728  info << "# refinePeriodic START" << std::endl;
729 
730  auto X = makeC1Segment(x);
731  auto solver = makeC1Solver();
732  auto pm = makeC1PoincareMap(solver, section);
733 
734  auto vsec = section(X);
735  info << "# section(X) = " << vsec << " (should be small)" << std::endl;
736 
737  capd::vectalg::EuclLNorm<Vector, Matrix> euclNorm;
738  Real reachTime;
739  Vector Px, fPx, dx(X.storageDimension());
740  Matrix Id(X.storageDimension(), X.storageDimension()); Id.setToIdentity();
741  pm.setInitialV(X);
742  C1Solution PX(X); PX *= 0.;
743  pm(X, PX, reachTime, x, Px, fPx, V, DP);
744 
745  double diff = euclNorm(Px-x);
746  info << "# reachTime = " << reachTime << std::endl;
747  info << "# ||PX - X|| = " << diff << std::endl;
748  info << "# Solving for Newton..." << std::flush;
749  capd::matrixAlgorithms::gauss(DP-Id, x-Px, dx);
750  info << "# DONE!" << std::endl;
751  x += dx;
752  info << "# refinePeriodic END" << std::endl;
753  return diff;
754  }
766  Real refinePeriodic(std::ostream& info, JetSection& section, Solution& X, Matrix& V, Matrix&DP){
767  auto x = X.get_x();
768  auto jacSection = makeC1Section(Vector(section), section.get_c());
769  auto diff = refinePeriodic(info, jacSection, x, V, DP);
770  X.set_x(x);
771  return diff;
772  }
