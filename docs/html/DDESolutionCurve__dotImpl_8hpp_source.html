47 template<typename SetSpec>
48 typename DDESolutionCurve<SetSpec>::ScalarType
49 DDESolutionCurve<SetSpec>::dot(DDESolutionCurve<SetSpec>::VectorType const& v) const{
50  // TODO: (NOT URGENT) dim check?
51  size_type vd = v.dimension();
52  size_type N0 = storageN0();
53  size_type d = dimension();
54  MatrixType CT(N0, vd);
55  MatrixType BT(vd, vd);
56  VectorType r(vd);
57  VectorType x(vd);
58  MatrixType cC = m_valueAtCurrent.get_C();
59  MatrixType cB = m_valueAtCurrent.get_B();
60  VectorType cr = m_valueAtCurrent.get_r();
61  VectorType cx = m_valueAtCurrent.get_x();
62  size_type I = 0;
63  size_type J = 0;
64  for (size_type i = 0; i < d && I < vd; ++i, ++I){
65  x[J + i] = cx[i];
66  r[J + i] = cr[i];
67  for (size_type j = 0; j < N0; ++j)
68  CT[j][J + i] = cC[i][j];
69  for (size_type j = 0; j < d; ++j)
70  BT[J + j][J + i] = cB[i][j];
71  }
72  J += d;
73  for (auto ijet = rbegin(); ijet != rend() && I < vd; ++ijet){
74  for (auto icoeff = (*ijet)->beginJet(); icoeff != (*ijet)->endJet() && I < vd; ++icoeff){
75  cC = icoeff->get_C();
76  cB = icoeff->get_B();
77  cr = icoeff->get_r();
78  cx = icoeff->get_x();
79  for (size_type i = 0; i < d && I < vd; ++i, ++I){
80  x[J + i] = cx[i];
81  r[J + i] = cr[i];
82  for (size_type j = 0; j < N0; ++j)
83  CT[j][J + i] = cC[i][j];
84  for (size_type j = 0; j < d; ++j)
85  BT[J + j][J + i] = cB[i][j];
86  }
87  J += d;
88  }
89  }
90  return v * x + (CT * v) * (*m_r0) + (BT * v) * r;
91 }

93 template<typename SetSpec>
94 template<typename JetSection>
95 typename DDESolutionCurve<SetSpec>::ScalarType
96 DDESolutionCurve<SetSpec>::dot(JetSection const& v) const{
97  // TODO: (NOT URGENT) dim check?
98  size_type vd = v.storageDimension();
99  size_type N0 = storageN0();
100  size_type d = dimension();
101  MatrixType CT(N0, vd);
102  MatrixType BT(vd, vd);
103  VectorType r(vd);
104  VectorType x(vd);
105  MatrixType cC = m_valueAtCurrent.get_C();
106  MatrixType cB = m_valueAtCurrent.get_B();
107  VectorType cr = m_valueAtCurrent.get_r();
108  VectorType cx = m_valueAtCurrent.get_x();
109  size_type I = 0;
110  size_type J = 0;
111  for (size_type i = 0; i < d && I < vd; ++i, ++I){
112  x[J + i] = cx[i];
113  r[J + i] = cr[i];
114  for (size_type j = 0; j < N0; ++j)
115  CT[j][J + i] = cC[i][j];
116  for (size_type j = 0; j < d; ++j)
117  BT[J + j][J + i] = cB[i][j];
118  }
119  J += d;
120  auto vjet = v.rbegin();
121  auto cjet = rbegin();
122  for (; cjet != rend() && vjet != v.rend(); ++cjet, ++vjet){
123  size_type order = vjet->order();
124  auto icoeff = (*cjet)->beginJet();
125  for (size_type k = 0; k <= order && I < vd; ++icoeff, ++k){
126  if (icoeff == (*cjet)->endJet())
127  throw std::logic_error("DDESolutionCurve::dot(JetSection): curve has not enough order at some jet to match the section.");
128  cC = icoeff->get_C();
129  cB = icoeff->get_B();
130  cr = icoeff->get_r();
131  cx = icoeff->get_x();
132  for (size_type i = 0; i < d && I < vd; ++i, ++I){
133  x[J + i] = cx[i];
134  r[J + i] = cr[i];
135  for (size_type j = 0; j < N0; ++j)
136  CT[j][J + i] = cC[i][j];
137  for (size_type j = 0; j < d; ++j)
138  BT[J + j][J + i] = cB[i][j];
139  }
140  J += d;
141  }
142  }
143  VectorType vv(v);
144  return vv * x + (CT * vv) * (*m_r0) + (BT * vv) * r;
145 }
