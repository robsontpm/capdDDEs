70 template<typename ScalarType>
71 std::pair<SetRelation, std::string> checkRelation(ScalarType const& a, ScalarType const& b){
72  std::string op = "";
73  SetRelation rel = SetRelation::unkn;
74  if (a.contains(b)){
75  op = "suPset";
76  rel = SetRelation::suPset;
77  } else if (b.contains(a)){
78  op = "suBset";
79  rel = SetRelation::suBset;
80  } else if (a.leftBound() > b.rightBound()) {
81  op = "miss R";
82  rel = SetRelation::miss_R;
83  } else if (b.leftBound() > a.rightBound()) {
84  op = "miss L";
85  rel = SetRelation::miss_L;
86  } else if (a.contains(b.leftBound())) {
87  op = "int L ";
88  rel = SetRelation::int_L;
89  } else if (a.contains(b.rightBound())) {
90  op = "int R ";
91  rel = SetRelation::int_R;
92  } else {
93  op = "unknown (should not happen)";
94  rel = SetRelation::unkn;
95  }
96  return std::make_pair(rel, op);
97 }

109 template<typename VectorSpec>
110 class DDECompareHelper {
111 public:
112  typedef VectorSpec VectorType;
113  typedef typename VectorType::size_type size_type;
114  typedef typename VectorType::ScalarType ScalarType;
115 
116  DDECompareHelper(VectorType const& orig, VectorType const& other): m_orig(orig), m_other(other), m_counter(7, 0) {
117  for (size_type i = 0; i < m_orig.dimension(); ++i){ // TODO: use iterators
118  auto relation = checkRelation(m_other[i], m_orig[i]);
119  ++m_counter[relation.first];
120  m_relations.push_back(relation);
121  }
122  }
123 
124  void printRelation(int i, std::ostream &out) const {
125  out << m_relations[i].second << " at " << i << std::endl;
126  out << " " << printInterval(m_other[i]) << std::endl;
127  out << " " << m_relations[i].second << std::endl;
128  out << " " << printInterval(m_orig[i]) << std::endl;
129  {
130  ScalarType a1, a2, r1, r2;
131  m_other[i].split(a1, r1);
132  m_orig[i].split(a2, r2);
133  if (r1.rightBound() > r2.rightBound()){ auto tmp = r2; r2 = r1; r1 = tmp; }
134  std::ostringstream margin; margin.precision(10);
135  margin << (1.0 - (r1.rightBound() / r2.rightBound()));
136  if (r2.rightBound() != 0.){
137  out << " (diameter margin: " << margin.str() << ")" << std::endl;
138  }
139  }
140  out << std::endl;
141  }
