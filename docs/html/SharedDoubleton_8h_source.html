319  BaseClass& add(VectorType const & v){
320  size_type d = this->dimension();
321  if (d != v.dimension()) throw std::logic_error("SharedDoubleton::add(vector): Input vector has incompatible dimension.");
322  assureOwner(OWN_x | OWN_r);
323  (*m_x) += v; VectorType s(v.dimension());
324  capd::vectalg::split(*m_x, s);
325  (*m_r) += (*m_Binv) * s;
326  return *this;
327  }
329  BaseClass& add(BaseClass const & set){
330  size_type N0 = this->storageN0();
331  size_type d = this->dimension();
332  if (set.storageN0() != N0) throw std::logic_error("SharedDoubleton::add(set): Sets has incompatible N0 dimensions.");
333  if (set.dimension() != d) throw std::logic_error("SharedDoubleton::add(set): Sets has incompatible dimensions.");
334  VectorType other_r0 = set.get_r0();
335  if (set.common_r0(m_r0)){
336  // testing by set.common_r() virtual gives advantage, that we do not need to know underlyings of set.
337  // if set is of the good class type, i.e. SharedDoubleton, then we have optimal situation.
338  assureOwner(OWN_x | OWN_r | OWN_C);
339  }else{
340  assureOwner(OWN_x | OWN_r | OWN_C | OWN_r0);
341  capd::vectalg::intervalHull(*m_r0, other_r0, *m_r0);
342  }
343  (*m_C) += set.get_C(); MatrixType S(d, N0);
344  capd::vectalg::split(*m_C, S);
345 
346  (*m_x) += set.get_x(); VectorType s(d);
347  capd::vectalg::split(*m_x, s);
348  (*m_r) += ((*m_Binv) * set.get_B()) * set.get_r() + (*m_Binv) * S * (*m_r0) + (*m_Binv) * s;
349  return *this;
350  }
352  BaseClass& mul(ScalarType const & c){
353  assureOwner(OWN_x | OWN_C | OWN_r);
354  size_type d = this->dimension();
355  size_type N0 = this->storageN0();
356  VectorType s(d);
357  MatrixType S(d, N0);
358  (*m_x) *= c;
359  capd::vectalg::split(*m_x, s);
360  (*m_C) *= c;
361  capd::vectalg::split(*m_C, S);
362  (*m_r) *= c;
363  (*m_r) += ((*m_Binv) * S) * (*m_r0) + (*m_Binv) * s;
364  return *this;
365  }

462  void updateBinv(){
463  assureOwner(OWN_Binv);
464  if (this->m_B){
465  MatrixType old_B = *(this->m_B);
466  this->Policy::computeBinvB(*(this->m_B), *(this->m_Binv), (*this->m_r));
467  // we had part old_B * m_r, but we want to have (new) m_B * new_r enclose the old part
468  // note that Policy::computeBinvB can alter both m_B and m_invB
469  // if m_B was in a good shape before computing invB then we do not need to correct m_r
470  if (this->m_r){
471  if (old_B != *this->m_B)
472  (*this->m_r) = ((*this->m_Binv) * old_B) * (*this->m_r);
473  }
474  } else {
475  throw std::logic_error("SharedDoubleton::updateBinv(): called, but matrix B not set!");
476  }

482  void rawSetup(
483  size_type dim, size_type N0,
484  const VectorType* x,
485  const MatrixType* C,
486  const VectorType* r0,
487  const MatrixType* B,
488  const VectorType* r,
489  const MatrixType* Binv,
490  OwnershipType ownership){
491  if (!x) ownership |= OWN_x;
492  if (!C) ownership |= OWN_C;
493  if (!r0) ownership |= OWN_r0;
494  if (!B) ownership |= OWN_B;
495  if (!r) ownership |= OWN_r;
496  if (!Binv) ownership |= OWN_Binv;
497  allocate(dim, N0, ownership);
498  if (x){ if (ownership[OWNERBIT_x]) *m_x = *x; else m_x = const_cast<VectorType*>(x); }
499  if (C){ if (ownership[OWNERBIT_C]) *m_C = *C; else m_C = const_cast<MatrixType*>(C); }
500  if (r0){ if (ownership[OWNERBIT_r0]) *m_r0 = *r0; else m_r0 = const_cast<VectorType*>(r0); }
501  if (B){ if (ownership[OWNERBIT_B]) *m_B = *B; else m_B = const_cast<MatrixType*>(B); }
502  if (r){ if (ownership[OWNERBIT_r]) *m_r = *r; else m_r = const_cast<VectorType*>(r); }
503  // r must be set before Binv for updateBinv() to work properly
504  if (Binv){
505  if (ownership[OWNERBIT_Binv]) *m_Binv = *Binv; else m_Binv = const_cast<MatrixType*>(Binv);
506  } else {
507  updateBinv();
508  }
509  m_owner = ownership;
510  }
512  void assureOwner(OwnershipType const & what = OWN_ALL){
513  // we assign to tmp all elements for simplicity, this des not cost much.
514  VectorType* tmp_x = m_x;
515  MatrixType* tmp_C = m_C;
516  VectorType* tmp_r0 = m_r0;
517  MatrixType* tmp_B = m_B;
518  VectorType* tmp_r = m_r;
519  MatrixType* tmp_Binv = m_Binv;
520  // Below: to_allocate will give the bits where what is 1 and m_owner is 0
521  // please check this! Those are elements that needs to be allocated
522  // because they were not owned by the object and we want them to be owned
523  // (wee need to copy them by value to newly allocated space)
524  // other items are either not required to be owned (0 in what)
525  // or are already owned (1 in m_owner).
526  // ~(~what | m_owner) = 1 <=> what = 1, owner = 0
527  OwnershipType to_allocate = ~(~what | m_owner);
528  // below: allocate this is safe, as it will only perform new on the
529  // elements that are already owned by someone else (0 in m_owner)
530  allocate(dimension(), storageN0(), to_allocate);
531  // now we copy the data from those vectors that needs to be copied
532  if (to_allocate[OWNERBIT_x]) *m_x = *tmp_x;
533  if (to_allocate[OWNERBIT_C]) *m_C = *tmp_C;
534  if (to_allocate[OWNERBIT_r0]) *m_r0 = *tmp_r0;
535  if (to_allocate[OWNERBIT_B]) *m_B = *tmp_B;
536  if (to_allocate[OWNERBIT_r]) *m_r = *tmp_r;
537  if (to_allocate[OWNERBIT_Binv]) *m_Binv = *tmp_Binv;

558  throw std::domain_error(info.str());

563  info << "was " << dimension() << ", expected " << m_Binv->numberOfRows() << ".";
564  throw std::domain_error(info.str());
565  }
566  if (dimension() != m_C->numberOfRows()) { throw std::domain_error("SharedDoubleton"+what+": C and x has incompatible dimensions."); }
567  if (dimension() != m_r->dimension()) { throw std::domain_error("SharedDoubleton"+what+": r has incompatible dimensions."); }
568  } catch (...){
569  // TODO: (NOT URGENT) maybe add more info to exception?
570 // std::cerr << (m_C ? *m_C : MatrixType()) << "|";

572  throw;

583  }
