68  VectorWithJacData& setMatrix(MatrixType const& D){
69  size_type d = this->dimension();
70  if (D.numberOfRows() != d)
71  throw std::logic_error("VectorWithJacData::setMatrix(): D has bad number of rows");
72  size_type I = 0;
73  m_Jac.clear();
74  while(I < D.numberOfColumns()){
75  MatrixSpec M(d, d);
76  // first go along columns (i, I),
77  // see that I changes same number as i, so after loop I is increased by d
78  // then we go with j along each row for each column.
79  // this way we partitions matrix D into matrices M of dim (d,d)
80  for (size_type i = 0; i < d && I < D.numberOfColumns(); ++i, ++I)
81  for (size_type j = 0; j < d; j++)
82  M[j][i] = D[j][I];
83  m_Jac.push_back(M);
84  }
85  return *this;
86  }

100  VectorWithJacData& operator*=(ScalarType const& c){
101  this->VectorSpec::operator*=(c);
102  for (auto i = m_Jac.begin(); i != m_Jac.end(); ++i) (*i) *= c;
103  return *this;

106  VectorWithJacData& operator+=(VectorWithJacData const& other){
107  this->VectorSpec::operator+=(other);
108  auto i = m_Jac.begin();
109  auto o = other.m_Jac.begin();
110  for (; i != m_Jac.end() && o != other.m_Jac.end(); ++i, ++o) (*i) += (*o);
111  return *this;
112  }

114  operator VectorSpec() { return *this; }
115  operator MatrixSpec() {
116  size_type d = this->dimension();
117  size_type N0 = d * m_Jac.size();
118  size_type I = 0;
119  MatrixSpec M(d, N0);
120  for (auto jaci = m_Jac.begin(); jaci != m_Jac.end(); ++jaci)
121  for (size_type i = 0; i < d; ++i, ++I)
122  for (size_type j = 0; j < d; ++j)
123  M[j][I] = (*jaci)[j][i];
124  return M;
125  }

194  void operator()(CurveType& in_out_curve, JacobianStorageType& D){
195  // TODO: DRY (see other)
196  auto& curve = in_out_curve;
197 
198  TimePointType t0 = curve.t0();
199  TimePointType th = t0 + curve.getStep();
200  RealType h = curve.getStep();
201 
202  size_type d = m_map.imageDimension();
203  VectorType zero(d);
204  MatrixType Zero(d, d);
205 
206  // collecting computation data
207  size_type coeffs_order = m_maxOrder;
208  VariableStorageType u;
209  m_map.collectComputationData(t0, th, h, curve, u, coeffs_order);
210  if (coeffs_order > m_maxOrder) coeffs_order = m_maxOrder;
211  ValueStorageType v; FunctionalMapType::convert(u, v);
212 
213  ValueStorageType Phi_coeffs_t0(coeffs_order + 1);
214  ValueStorageType Phi_z(1); Phi_z[0] = zero;
215  JacobianStorageType JacPhi_z(1); JacPhi_z[0].resize(u.size(), Zero);
216  JacobianStorageType JacPhi_coeffs_t0(coeffs_order + 1);
217  for (auto jaci = JacPhi_coeffs_t0.begin(); jaci != JacPhi_coeffs_t0.end(); ++jaci)
218  jaci->resize(u.size(), Zero);
219 
220  this->oneStep(t0, th, h, v, Phi_coeffs_t0, JacPhi_coeffs_t0, Phi_z, JacPhi_z);
221 
222  // FunctionalMapType::deconvert(Phi_coeffs_t0, new_jet);
223  typedef typename DataType::MatrixStorageType MatrixStorageType;
224  VariableStorageType new_jet;
225  // IMPORTANT: we assume that other coeffs (PhiCoeffs) are dependent on the same list of variables as the z(t)
226  size_type varCount = u[0].getMatrixData().size();
227  for (size_type k = 0; k <= coeffs_order; ++k){
228  MatrixStorageType DJet_k(varCount, Zero);
229  for (size_type i = 0; i < varCount; ++i)
230  for (size_type j = 0; j < u.size(); ++j)
231  DJet_k[i] += JacPhi_coeffs_t0[k][j] * u[j].getMatrixData()[i];
232 
233  new_jet.push_back(DataType(Phi_coeffs_t0[k], DJet_k));
234  if (k > 0) D.push_back(DJet_k);
235  }
236  curve.addPiece(new JetType(th, new_jet));
237 
238  MatrixStorageType Dz(varCount, Zero);
239  for (size_type i = 0; i < varCount; ++i)
240  for (size_type j = 0; j < u.size(); ++j)
241  Dz[i] += JacPhi_z[0][j] * u[j].getMatrixData()[i];
242 
243  curve.setValueAtCurrent(DataType(Phi_z[0], Dz));
244  D.push_back(Dz);
245  }

259  void oneStep(
260  TimePointType const& in_t0,
261  TimePointType const& in_th,
262  RealType const& in_h,
263  ValueStorageType const& in_u,
264  ValueStorageType& out_Phi_coeffs_t0,
265  JacobianStorageType& out_JacPhi_coeffs_t0,
266  ValueStorageType& out_Phi_z, // for possible future use, I assume that there might be more than only value of x(t+h) !
267  JacobianStorageType& out_JacPhi_z) // I need storage for as many as there will be entries in out_Phi_z
268  {
269  auto& h = in_h;
270  auto& u = in_u;
271 
272  // compute both value and Jacobian (in nonrigorous version u is a point vector!)
273  m_map.computeDDECoefficients(in_t0, u, out_Phi_coeffs_t0, out_JacPhi_coeffs_t0);
274  // now we have Taylor coefficients at t0, we now simply use standard
275  // Taylor method, as in case of ODEs
276  // we explicitely eval jet at t0 as a Taylor series to assure that the
277  // jacobian computed along is valid.
278  // TODO: (NOT URGENT) rewrite as iterators?
279  size_type k = out_Phi_coeffs_t0.size() - 1;
280  while(true){
281  out_Phi_z[0] = out_Phi_coeffs_t0[k] + h * out_Phi_z[0];
282  for (size_type j = 0; j < u.size(); ++j){
283  out_JacPhi_z[0][j] = out_JacPhi_coeffs_t0[k][j] + h * out_JacPhi_z[0][j];
284  }
285  if (k == 0) break; // this prevents from infinite loop in case size_type is unsigned
286  --k;
287  }
288 
289  // and we are done!
290  }
